013    /**
014     * Provides methods to query KiSAO.
015     * Example:
016     * <pre>
017     * {@code
018     * // Create KiSAOQueryMaker instance, which uses last version of kisao.owl ontology
019     * // (URL: http://biomodels.net/kisao/KISAO).
020     * IKiSAOQueryMaker kisaoQuery = new KiSAOQueryMaker();
021     * // To use kisao.owl, stored locally, instead, one should specify IRI constructor argument:
022     * // IKiSAOQueryMaker kisaoQuery = new KiSAOQueryMaker(IRI.create("file:///..."));
023     *
024     * // Get KiSAO element IRI by name...
025     * IRI iri = kisaoQuery.searchByName("tau-leaping method").iterator().next();
026     * // ... or by MIRIAM URN ...
027     * assert iri.equals(kisaoQuery.searchById("urn:miriam:biomodels.kisao:KISAO_0000039"));
028     * // ... or by ID ...
029     * assert iri.equals(kisaoQuery.searchById("kisao:0000039"));
030     *
031     * // Check if it's an algorithm
032     * assert kisaoQuery.isAlgorithm(iri);
033     * // Get algorithm name
034     * String name = kisaoQuery.getName(iri);
035     * System.out.printf("%s (%s)\n", name, iri);
036     * // Get other names
037     * for (String synonym : kisaoQuery.getAllSynonyms(iri)) {
038     *      System.out.printf(" is also known as %s\n", synonym);
039     * }
040     * // Get definition
041     * System.out.printf(" which is %s\n", kisaoQuery.getDefinition(iri));
042     * // Get references
043     * for (Map.Entry<String, String> entry : kisaoQuery.getLinks(iri).entrySet()) {
044     *      System.out.printf(" (see also: %s -- %s)\n", entry.getKey(), entry.getValue());
045     * }
046     *
047     * System.out.println("");
048     * // Check if algorithm has characteristics discrete variable and stochastic behaviour
049     * assert kisaoQuery.hasCharacteristic(iri, true, KiSAOIRI.STOCHASTIC_SYSTEM_BEHAVIOUR_CHARACTERISTIC_IRI,
050     *      KiSAOIRI.DISCRETE_VARIABLE_CHARACTERISTIC_IRI);
051     * // Get positive algorithm characteristics,
052     * // such as: 'has characteristic' some 'continuous variable'
053     * for (IRI characteristic : kisaoQuery.getCharacteristics(iri, true)) {
054     *      System.out.printf(" has characteristic %s\n", kisaoQuery.getName(characteristic));
055     * }
056     * // Get negative algorithm characteristics,
057     * // such as: not 'has characteristic' some 'spatial description'
058     * for (IRI characteristic : kisaoQuery.getCharacteristics(iri, false)) {
059     *      System.out.printf(" hasn't characteristic %s\n", kisaoQuery.getName(characteristic));
060     * }
061     *
062     * System.out.println("");
063     * // Get parameters
064     * for (IRI parameter : kisaoQuery.getParameters(iri)) {
065     *      System.out.printf(" has parameter %s of type %s\n", kisaoQuery.getName(parameter),
066     *          kisaoQuery.getParameterType(parameter).getName());
067     * }
068     * // Get ancestors (true means that only direct ancestors are included)
069     * for (IRI ancestor : kisaoQuery.getAncestors(iri, true)) {
070     *      System.out.printf(" is a %s\n", kisaoQuery.getName(ancestor));
071     * }
072     *
073     * System.out.println("");
074     * // Get descendants (false means that indirect descendants are also included)
075     * for (IRI descendant : kisaoQuery.getDescendants(iri, false)) {
076     *      System.out.printf(" is specified in %s\n", kisaoQuery.getName(descendant));
077     * }
078     *
079     * System.out.println("\n");
080     * // Get all the algorithms using deterministic rules and continuous variables
081     * for (IRI algorithm : kisaoQuery.getAlgorithmsByCharacteristic(true,
082     *          KiSAOIRI.DETERMINISTIC_SYSTEM_BEHAVIOUR_CHARACTERISTIC_IRI, KiSAOIRI.CONTINUOUS_VARIABLE_CHARACTERISTIC_IRI,
083     *          KiSAOIRI.PROGRESSION_WITH_FIXED_TIME_STEP_CHARACTERISTIC_IRI)) {
084     *      System.out.printf("%s is continuous deterministic and has fixed time steps\n", kisaoQuery.getName(algorithm));
085     * }
086     *
087     * System.out.println("\n");
088     * // Make our own query
089     * OWLReasoner reasoner = kisaoQuery.getReasoner();
090     * OWLDataFactory dataFactory = kisaoQuery.getDataFactory();
091     * System.out.printf("%s and %s are%s disjoint\n",
092     * kisaoQuery.getName(KiSAOIRI.KINETIC_SIMULATION_ALGORITHM_IRI),
093     * kisaoQuery.getName(KiSAOIRI.KINETIC_SIMULATION_ALGORITHM_CHARACTERISTIC_IRI),
094     * reasoner.isEntailed(dataFactory.getOWLDisjointClassesAxiom(
095     *      dataFactory.getOWLClass(KiSAOIRI.KINETIC_SIMULATION_ALGORITHM_IRI),
096     *      dataFactory.getOWLClass(KiSAOIRI.KINETIC_SIMULATION_ALGORITHM_CHARACTERISTIC_IRI))) ?
097     *          "" : " not");
098     *
099     * System.out.println("\n");
100     * //Check which parameters has every deterministic algorithm
101     * for (IRI parameterIRI : kisaoQuery.getParametersByCharacteristic(true, KiSAOIRI.DETERMINISTIC_SYSTEM_BEHAVIOUR_CHARACTERISTIC_IRI)) {
102     *      System.out.printf("every deterministic algorithm uses parameter %s\n",
103     *          kisaoQuery.getName(parameterIRI));
104     * }
105     * }
106     * </pre>
107     *
108     * @author Anna Zhukova
109     *         Date: 13-May-2011
110     *         Time: 10:20:20
111     */
112    
113    public interface IKiSAOQueryMaker {
114    
115        /**
116         * Provides a set of IRIs of simulation algorithms, stored in KiSAO.
117         *
118         * @return set of algorithm IRIs.
119         */
120        Set<IRI> getAllAlgorithms();
121    
122        /**
123         * Provides a set of IRIs of simulation algorithm characteristics, stored in KiSAO.
124         * Subsumption characteristics (e.g. 'type of time steps') are not included.
125         *
126         * @return set of algorithm characteristic IRIs.
127         */
128        Set<IRI> getAllCharacteristics();
129    
130        /**
131         * Provides a set of IRIs of simulation algorithm parameters, stored in KiSAO.
132         * Subsumption parameters (e.g. 'granularity control parameter') are not included.
133         *
134         * @return set of algorithm parameter IRIs.
135         */
136        Set<IRI> getAllParameters();
137    
138        /**
139         * Returns name (rdfs:label) by given KiSAO IRI.
140         *
141         * @param iri KiSAO IRI.
142         * @return name of the KiSAO entity.
143         */
144        String getName(IRI iri);
145    
146        /**
147         * Returns definition (rdfs:comment) by given KiSAO IRI.
148         *
149         * @param iri KiSAO IRI.
150         * @return definition of the KiSAO entity.
151         */
152        String getDefinition(IRI iri);
153    
154        /**
155         * Returns whether the KiSAO entry with the given IRI is deprecated or not.
156         *
157         * @param iri KiSAO IRI.
158         * @return true, if the KiSAO entry with the given IRI is deprecated.
159         */
160        boolean isDeprecated(IRI iri);
161    
162        /**
163         * Returns synonyms for the KiSAO entry with the given IRI.
164         *
165         * @param iri KiSAO IRI.
166         * @return set of synonyms.
167         */
168        Set<String> getAllSynonyms(IRI iri);
169    
170        /**
171         * Returns synonyms of the specified type for the KiSAO entry with the given IRI.
172         *
173         * @param iri  KiSAO IRI.
174         * @param type type of the synonyms to look for.
175         * @return set of synonyms.
176         * @see net.biomodels.kisao.SynonymType
177         */
178        Set<String> getSynonyms(IRI iri, SynonymType type);
179    
180        /**
181         * Returns links to the articles/books, describing the KiSAO entry with the given IRI.
182         *
183         * @param iri KiSAO IRI.
184         * @return map article-location (usually, a MIRIAM URN) -> article-name.
185         */
186        Map<String, String> getLinks(IRI iri);
187    
188        /**
189         * Given a KiSAO IRI of a simulation algorithm,
190         * returns a collection of IRIs of characteristics, it possess.
191         *
192         * @param algorithmIri KiSAO IRI of a simulation algorithm.
193         * @param positive     false, if negated characteristics should be looked at,
194         *                     for example 'KMC' not ('has characteristic' 'spatial description').
195         * @param type         type (optional) IRIs of the characteristic categories (e.g. 'system behaviour') to be considered.
196         *                     All the characteristics are considered if no type is specified.
197         * @return set of IRIs of the algorithm characteristics.
198         */
199        Set<IRI> getCharacteristics(IRI algorithmIri, boolean positive, IRI... type);
200    
201        /**
202         * Given the class expression, describing potential algorithm,
203         * returns a collection of IRIs of characteristics, it possess.
204         *
205         * @param algorithm    OWLClassExpression describing algorithm.
206         * @param positive     false, if negated characteristics should be looked at,
207         *                     for example 'KMC' not ('has characteristic' 'spatial description').
208         * @param type         type (optional) IRIs of the characteristic categories (e.g. 'system behaviour') to be considered.
209         *             All the characteristics are considered if no type is specified.
210         * @return set of IRIs of the algorithm characteristics.
211         */
212        Set<IRI> getCharacteristics(OWLClassExpression algorithm, boolean positive, IRI... type);
213    
214        /**
215         * Given KiSAO IRIs of simulation algorithm characteristics,
216         * returns a collection of IRIs of simulation algorithms, which have them.
217         *
218         * @param characteristicIri KiSAO IRIs of simulation algorithm characteristics.
219         * @param positive          false, if negated characteristics should be looked at,
220         *                          for example 'KMC' not ('has characteristic' 'spatial description').
221         * @return set of IRIs of algorithms, which have the characteristic.
222         */
223        Set<IRI> getAlgorithmsByCharacteristic(boolean positive, IRI... characteristicIri);
224    
225        /**
226         * Checks if the algorithm with the specified KiSAO IRI 'has characteristic' (or not 'has characteristic', if positive is false)
227         * with the specified IRIs.
228         * Example:
229         * <pre>
230         * {@code
231         * IKiSAOQueryMaker query = ...
232         * IRI lsoda = IRI.create("http://www.biomodels.net/kisao/KISAO#KISAO_0000094");
233         * // false specifies that we check a negated axiom: "LSODA not ('has characteristic' some 'spatial description')"
234         * assert true ==  query.hasCharacteristic(lsoda, KiSAO.SPATIAL_DESCRIPTION_CHARACTERISTIC_IRI, false);
235         * // true specifies that we check a normal axiom: "LSODA 'has characteristic' some 'continuous variable'"
236         * assert true ==  query.hasCharacteristic(lsoda, KiSAO.CONTINUOUS_VARIABLE_CHARACTERISTIC_IRI, true);
237         * }
238         * </pre>
239         *
240         * @param algorithmIRI      KiSAO IRI of the simulation algorithm.
241         * @param characteristicIRI KiSAO IRIs of the simulation algorithm characteristics.
242         * @param positive          false, if negated characteristics should be looked at.
243         * @return true, if the algorithms with the specified KiSAO IRI 'has characteristic' (or not 'has characteristic', if positive is false)
244         *         with the specified IRI.
245         */
246        boolean hasCharacteristic(IRI algorithmIRI, boolean positive, IRI... characteristicIRI);
247    
248        /**
249         * Returns a set of parameters that are used by the specified algorithm.
250         *
251         * @param algorithm KiSAO IRI of the algorithm.
252         * @return set of parameter KiSAO IRIs.
253         */
254        Set<IRI> getParameters(IRI algorithm);
255    
256        /**
257         * Returns a set of parameters that are used by the class expression, describing potential algorithm.
258         * Example:
259         * <pre>
260         * {@code
261         * OWLDataFactory dataFactory = kisaoQueryMaker.getDataFactory();
262         * OWLClassExpression algorithm = dataFactory.getOWLObjectSomeValuesFrom(dataFactory.getOWLObjectProperty(HAS_CHARACTERISTIC_IRI),
263         * dataFactory.getOWLClass(DETERMINISTIC_SYSTEM_BEHAVIOUR_CHARACTERISTIC_IRI));
264         * // parameters of methods with deterministic behaviour
265         * for (IRI parameter : kisaoQueryMaker.getParameters(algorithm)) {
266         * System.out.println(kisaoQueryMaker.getName(parameter));
267         * }
268         * }
269         * </pre>
270         *
271         * @param algorithm OWLClassExpression describing potential algorithm.
272         * @return set of parameter KiSAO IRIs.
273         */
274        Set<IRI> getParameters(OWLClassExpression algorithm);
275    
276        /**
277         * Returns a set of parameters that are used by the algorithm, which have specified characteristic(s).
278         *
279         * @param characteristicIRI characteristic KiSAO IRIs.
280         * @param positive          false, if negated characteristics should be looked at.
281         * @return set of parameter KiSAO IRIs.
282         */
283        Set<IRI> getParametersByCharacteristic(boolean positive, IRI... characteristicIRI);
284    
285        /**
286         * Returns a set of parameters that are used by the algorithm, which have specified characteristic(s)
287         * and is descendant of the specified algorithm.
288         *
289         * @param ancestor          ancestor algorithm KiSAO IRI.
290         * @param characteristicIRI characteristic KiSAO IRIs.
291         * @param positive          false, if negated characteristics should be looked at.
292         * @return set of parameter KiSAO IRIs.
293         */
294        Set<IRI> getParametersByAncestorAndCharacteristic(IRI ancestor, boolean positive, IRI... characteristicIRI);
295    
296        /**
297         * Returns a set of parameters that are used by the algorithm, which have specified characteristic(s)
298         * and is descendant of the specified algorithm.
299         *
300         * @param ancestor          OWLClassExpression describing ancestor algorithm.
301         * @param characteristicIRI characteristic KiSAO IRIs.
302         * @param positive          false, if negated characteristics should be looked at.
303         * @return set of parameter KiSAO IRIs.
304         */
305        Set<IRI> getParametersByAncestorAndCharacteristic(OWLClassExpression ancestor, boolean positive, IRI... characteristicIRI);
306    
307        /**
308         * Checks if the algorithm with the specified KiSAO IRI 'has parameter' with the specified IRIs.
309         * Example:
310         * <pre>
311         * {@code
312         * IKiSAOQueryMaker kisaoQueryMaker = ...
313         * IRI lsoda = IRI.create("http://www.biomodels.net/kisao/KISAO#KISAO_0000094");
314         * IRI stepSize = IRI.create("http://www.biomodels.net/kisao/KISAO#KISAO_0000114");
315         * assert true ==  kisaoQueryMaker.hasParameter(lsoda, stepSize);
316         * }
317         * </pre>
318         *
319         * @param algorithmIRI KiSAO IRI of the simulation algorithm.
320         * @param parameterIRI KiSAO IRIs of the simulation algorithm parameters.
321         * @return true, if the algorithms with the specified KiSAO IRI 'has parameter' with the specified IRI.
322         */
323        boolean hasParameter(IRI algorithmIRI, IRI... parameterIRI);
324    
325        /**
326         * Checks if the algorithm, described by given OWLClassExpression, 'has parameter' with the specified IRIs.
327         * Example:
328         * <pre>
329         * {@code
330         * IKiSAOQueryMaker kisaoQueryMaker = ...
331         * OWLDataFactory dataFactory = kisaoQueryMaker.getDataFactory();
332         * OWLClassExpression algorithm = dataFactory.getOWLObjectSomeValuesFrom(dataFactory.getOWLObjectProperty(HAS_CHARACTERISTIC_IRI),
333         * dataFactory.getOWLClass(DETERMINISTIC_SYSTEM_BEHAVIOUR_CHARACTERISTIC_IRI));
334         * IRI stepSize = IRI.create("http://www.biomodels.net/kisao/KISAO#KISAO_0000114");
335         * assert true ==  kisaoQueryMaker.hasParameter(algorithm, stepSize);
336         * }
337         * </pre>
338         *
339         * @param algorithm    OWLClassExpression describing algorithm.
340         * @param parameterIRI KiSAO IRIs of the simulation algorithm parameters.
341         * @return true, if the algorithms, described by given OWLClassExpression, 'has parameter' with the specified IRI.
342         */
343        boolean hasParameter(OWLClassExpression algorithm, IRI... parameterIRI);
344    
345        /**
346         * Checks if the algorithm, described by its characteristics, 'has parameter' with the specified IRIs.
347         *
348         * @param characteristicIRIs KiSAO IRIs of the characteristics.
349         * @param parameterIRI       KiSAO IRIs of the simulation algorithm parameters.
350         * @param positive           false, if negated characteristics should be looked at.
351         * @return true, if the algorithms, described by given characteristics, 'has parameter' with the specified IRI.
352         */
353        boolean hasParameter(boolean positive, IRI[] characteristicIRIs, IRI... parameterIRI);
354    
355        /**
356         * Given the KiSAO IRI of a parameter, returns parameter type, e.g. Double for 'absolute tolerance'.
357         *
358         * @param parameter KiSAO IRI of the parameter.
359         * @return Class specifying the parameter type.
360         */
361        Class getParameterType(IRI parameter);
362    
363        /**
364         * Returns type (algorithm, characteristic or parameter) of the KiSAO element with the given IRI.
365         *
366         * @param iri KiSAO IRI.
367         * @return KiSAOType
368         */
369        KiSAOType getType(IRI iri);
370    
371        /**
372         * Checks if the element with the specified IRI represents simulation algorithm in KiSAO.
373         *
374         * @param iri KiSAO IRI.
375         * @return true, if the element with the specified IRI represents simulation algorithm in KiSAO.
376         */
377        boolean isAlgorithm(IRI iri);
378    
379        /**
380         * Checks if the element with the specified IRI represents
381         * simulation algorithm characteristic in KiSAO.
382         *
383         * @param iri KiSAO IRI.
384         * @return true, if the element with the specified IRI represents
385         *         simulation algorithm characteristic in KiSAO.
386         */
387        boolean isCharacteristic(IRI iri);
388    
389        /**
390         * Checks if the element with the specified IRI represents
391         * simulation algorithm parameter in KiSAO.
392         *
393         * @param iri KiSAO IRI.
394         * @return true, if the element with the specified IRI represents
395         *         simulation algorithm parameter in KiSAO.
396         */
397        boolean isParameter(IRI iri);
398    
399        /**
400         * Looks for a KiSAO entry with the specified name or synonym.
401         *
402         * @param name String representing name or synonym.
403         * @return set of KiSAO IRI.
404         */
405        Set<IRI> searchByName(String name);
406    
407        /**
408         * Looks for a KiSAO entries matching specified query.
409         *
410         * @param query String containing entry identifier (e.g. kisao:0000001)
411         *        or MIRIAM URI (e.g. urn:miriam:biomodels.kisao:KISAO_0000001)
412         *        or URI (e.g. "http://www.biomodels.net/kisao/KISAO#KISAO_0000259").
413         * @return KiSAO IRI.
414         */
415        IRI searchById(String query);
416    
417        /**
418         * Returns a set of descendant IRIs for the specified entry.
419         *
420         * @param entryIRI KiSAO IRI.
421         * @param direct   Whether should look only for direct descendants.
422         * @return set of descendant IRIs.
423         */
424        Set<IRI> getDescendants(IRI entryIRI, boolean direct);
425    
426        /**
427         * Returns a set of descendant IRIs for the specified entry.
428         *
429         * @param entry  OWLClassExpression describing ancestor entry.
430         * @param direct Whether should look only for direct descendants.
431         * @return set of descendant IRIs.
432         */
433        Set<IRI> getDescendants(OWLClassExpression entry, boolean direct);
434    
435        /**
436         * Returns a set of ancestor IRIs for the specified entry.
437         *
438         * @param entryIRI KiSAO IRI.
439         * @param direct   Whether should look only for direct ancestors.
440         * @return set of ancestor IRIs.
441         */
442        Set<IRI> getAncestors(IRI entryIRI, boolean direct);
443    
444        /**
445         * Returns a set of ancestor IRIs for the specified entry.
446         *
447         * @param entry  OWLClassExpression describing descendant entry.
448         * @param direct Whether should look only for direct ancestors.
449         * @return set of ancestor IRIs.
450         */
451        Set<IRI> getAncestors(OWLClassExpression entry, boolean direct);
452    
453        /**
454         * Checks if a descendantCandidate entry is a descendant (concerning rdf:subClassOf relationship)
455         * of the ancestorCandidate entry in KiSAO.
456         *
457         * @param descendantCandidate KiSAO IRI of the possible descendant.
458         * @param ancestorCandidate   KiSAO IRI of the possible ancestor.
459         * @return true, if the descendantCandidate entry is a descendant (concerning rdf:subClassOf relationship)
460         *         of the ancestorCandidate entry.
461         */
462        boolean isA(IRI descendantCandidate, IRI ancestorCandidate);
463    
464        /**
465         * Checks if a descendantCandidate entry is a descendant (concerning rdf:subClassOf relationship)
466         * of the ancestorCandidate entry in KiSAO.
467         *
468         * @param descendantCandidate OWLClassExpression for descendant.
469         * @param ancestorCandidate   OWLClassExpression for ancestor.
470         * @return true, if the descendantCandidate entry is a descendant (concerning rdf:subClassOf relationship)
471         *         of the ancestorCandidate entry.
472         */
473        boolean isA(OWLClassExpression descendantCandidate, OWLClassExpression ancestorCandidate);
474    
475    
476        /**
477         * Returns miriam urn (urn:miriam:biomodels.kisao:KISAO_XXXXXXX) by KiSAO IRI.
478         *
479         * @param iri KiSAO IRI.
480         * @return string, representing miriam urn (urn:miriam:biomodels.kisao:KISAO_XXXXXXX)
481         */
482        String getMiriamURI(IRI iri);
483    
484        /**
485         * Returns id (kisao:XXXXXXX) by KiSAO IRI.
486         *
487         * @param iri KiSAO IRI.
488         * @return string, representing id (kisao:XXXXXXX)
489         */
490        String getId(IRI iri);
491    
492        /**
493         * Checks whether the algorithm with the specified IRI is a hybrid one.
494         *
495         * @param iri KiSAO IRI.
496         * @return if algorithm with the specified IRI is a hybrid one.
497         */
498        boolean isHybrid(IRI iri);
499    
500        /**
501         * Returns IRIs of the algorithms, the specified one is a hybrid of.
502         *
503         * @param iri KiSAO IRI of the hybrid algorithm.
504         * @return set of IRIs of the algorithms, the specified one is a hybrid of or null if it's not hybrid.
505         */
506        Set<IRI> getHybridOf(IRI iri);
507    
508        /**
509         * Checks whether the algorithm with the specified IRI is a complex one:
510         * is a hybrid one of uses other algorithms.
511         *
512         * @param iri KiSAO IRI.
513         * @return if algorithm with the specified IRI is a complex one.
514         */
515        boolean isComplex(IRI iri);
516    
517        /**
518         * Returns IRIs of the algorithms, the specified one uses.
519         *
520         * @param iri KiSAO IRI of the complex algorithm.
521         * @return set of IRIs of the algorithms, the specified one uses.
522         */
523        Set<IRI> getUsedAlgorithms(IRI iri);
524    
525        /**
526         * Returns IRIs of the algorithms, which use other algorithms.
527         *
528         * @return set of IRIs of complex algorithms.
529         */
530        Set<IRI> getComplexAlgorithms();
531    
532        /**
533         * Returns reasoner.
534         *
535         * @return OWLReasoner.
536         */
537        OWLReasoner getReasoner();
538    
539        /**
540         * Returns data factory.
541         *
542         * @return OWLDataFactory.
543         */
544        OWLDataFactory getDataFactory();
545    
546        /**
547         * Returns algorithms having the same characteristics as the specified one.
548         * @param algorithm The sample algorithm IRI.
549         * @param type (optional) IRIs of the characteristic categories (e.g. 'system behaviour') to be considered.
550         *             All the characteristics are considered if no type is specified.
551         * @return set of algorithm IRIs.
552         */
553        Set<IRI> getAlgorithmsWithSameCharacteristics(IRI algorithm, IRI... type);
554    
555        /**
556         * Returns algorithms having the same characteristics as the specified one.
557         * @param algorithm The sample algorithm IRI.
558         * @param n The number of algorithms to return. n &lt; 0 is considered as infinity.
559         * @param type (optional) IRIs of the characteristic categories (e.g. 'system behaviour') to be considered.
560         *             All the characteristics are considered if no type is specified.
561         * @return sorted (by similarity) list of n (or less if n were not found) IRIs of the algorithms most similar to the given one.
562         */
563        List<IRI> getNMostSimilarAlgorithms(IRI algorithm, int n, IRI... type);
564    
565        /**
566         * Distance between two algorithms in the hierarchy tree.
567         * @param algorithm1 IRI of the first algorithm.
568         * @param algorithm2 IRI of the second algorithm.
569         * @return distance value which is not less that 0 and not greater than 1.
570         */
571        double distance(IRI algorithm1, IRI algorithm2);
