042        /**
043         * Creates a new KiSAOQueryMaker.
044         *
045         * @param kisaoIri IRI for the KiSAO ontology.
046         *                 If kisaoIri == null, then default KiSAO_IRI will be used.
047         * @param factory  OWLReasonerFactory to specify which reasoner should be used.
048         *                 If factory == null, Hermit reasoner will be used.
049         * @param monitor  ReasonerProgressMonitor to be used to listen to the reasoner progress.
050         *                 If monitor == null, NullReasonerProgressMonitor will be used.
051         * @throws OWLOntologyCreationException If something is wrong with the ontology file/url.
052         * @see net.biomodels.kisao.impl.KiSAOQueryMaker KISAO_IRI
053         * @see org.semanticweb.HermiT.Reasoner
054         */
055        public KiSAOQueryMaker(IRI kisaoIri, OWLReasonerFactory factory, ReasonerProgressMonitor monitor) throws OWLOntologyCreationException {
056            OWLOntologyManager manager = OWLManager.createOWLOntologyManager();
057            dataFactory = manager.getOWLDataFactory();
058            if (kisaoIri == null) {
059                kisaoIri = KISAO_IRI;
060            }
061            kisao = manager.loadOntologyFromOntologyDocument(kisaoIri);
062            collectLabels();
063    
064            if (factory == null) {
065                factory = new Reasoner.ReasonerFactory();//new FaCTPlusPlusReasonerFactory();
066            }
067            if (monitor == null) {
068                monitor = new NullReasonerProgressMonitor();
069            }
070            reasoner = factory.createNonBufferingReasoner(kisao, new SimpleConfiguration(monitor));
071            addInferredAxioms(manager);
072        }

192            OWLClass clazz = dataFactory.getOWLClass(iri);
193            Set<String> result = new HashSet<String>();
194            Set<OWLAnnotation> synonym = clazz.getAnnotations(kisao,

205        public Set<String> getSynonyms(IRI iri, SynonymType type) {
206            assert type != null;
207            OWLClass clazz = dataFactory.getOWLClass(iri);
208            Set<String> result = new HashSet<String>();
209            Set<OWLAnnotation> synonym = clazz.getAnnotations(kisao,
210                    dataFactory.getOWLAnnotationProperty(SYNONYM_IRI));
211            OWLAnnotationProperty synonymType = dataFactory.getOWLAnnotationProperty(SYNONYM_TYPE_IRI);
212            for (OWLAnnotation name : synonym) {
213                if (type.equals(SynonymType.byName(getLiteral(getAnnotationAnnotation(iri, name, synonymType))))) {
214                    result.add(getLiteral(name));
215                }
216            }
217            return result;
218        }

255            OWLClassExpression query = getClassExpressionByCharacteristics(
256                    dataFactory.getOWLClass(KiSAOIRI.KINETIC_SIMULATION_ALGORITHM_IRI),

290        public Set<IRI> getDescendants(OWLClassExpression entry, boolean direct) {
291            if (entry == null) return null;
292            final Set<IRI> result = new HashSet<IRI>();
293            // not anonymous descendants
294            for (Node<OWLClass> clazz : reasoner.getSubClasses(entry, direct)) {
295                OWLClass descendant = clazz.getRepresentativeElement();
296                if (descendant != null && !descendant.isOWLNothing()) {
297                    result.add(descendant.getIRI());
298                }
299            }
300            return result;
301        }

307        public Set<IRI> getAncestors(OWLClassExpression entry, boolean direct) {
308            if (entry == null) return null;
309            final Set<IRI> result = new HashSet<IRI>();
310            // not anonymous ancestors
311            for (Node<OWLClass> clazz : reasoner.getSuperClasses(entry, direct)) {
312                OWLClass ancestor = clazz.getRepresentativeElement();
313                if (ancestor != null && !ancestor.isOWLThing()) {
314                    result.add(ancestor.getIRI());
315                }
316            }
317            return result;
318        }

320        public boolean isHybrid(IRI iri) {
321            OWLObjectProperty isHybrid = dataFactory.getOWLObjectProperty(IS_HYBRID_OF_IRI);
322            OWLClass ksa = dataFactory.getOWLClass(KINETIC_SIMULATION_ALGORITHM_IRI);
323            OWLObjectSomeValuesFrom someValuesFrom = dataFactory.getOWLObjectSomeValuesFrom(isHybrid, ksa);
324            OWLClass algorithm = dataFactory.getOWLClass(iri);
325            OWLClassExpression query = dataFactory.getOWLObjectIntersectionOf(algorithm, someValuesFrom);
326            return reasoner.getEquivalentClasses(query).contains(algorithm);
327        }

340        public Set<IRI> getParameters(OWLClassExpression algorithm) {
341            OWLObjectProperty hasParameter = dataFactory.getOWLObjectProperty(HAS_PARAMETER_IRI);
342            Set<IRI> result = new HashSet<IRI>();
343            for (Node<OWLClass> node : reasoner.getSubClasses(
344                    dataFactory.getOWLClass(KINETIC_SIMULATION_ALGORITHM_PARAMETER_IRI), false)) {
345                OWLClass element = node.getRepresentativeElement();
346                if (element.getSubClasses(kisao).isEmpty() && !element.isOWLNothing()) {
347                    OWLClassExpression query = dataFactory.getOWLObjectSomeValuesFrom(hasParameter, element);
348                    query = dataFactory.getOWLObjectIntersectionOf(algorithm, query);
349                    if (reasoner.isEntailed(dataFactory.getOWLEquivalentClassesAxiom(algorithm, query))) {
350                        result.add(element.getIRI());
351                    }
352                }
353            }
354            return result;
355        }

371        public boolean hasParameter(OWLClassExpression algorithmExpression, final IRI... parameterIRI) {
372            OWLObjectProperty hasParameter = dataFactory.getOWLObjectProperty(HAS_PARAMETER_IRI);
373            for (IRI iri : parameterIRI) {
374                OWLClassExpression query = algorithmExpression;
375                OWLClass parameter = dataFactory.getOWLClass(iri);
376                OWLObjectSomeValuesFrom someValuesFrom = dataFactory.getOWLObjectSomeValuesFrom(hasParameter, parameter);
377                query = dataFactory.getOWLObjectIntersectionOf(query, someValuesFrom);
378                if (!reasoner.isEntailed(dataFactory.getOWLEquivalentClassesAxiom(algorithmExpression, query))) {
379                    return false;
380                }
381            }
382            return true;
383        }

398        public Class getParameterType(IRI parameterIri) {
399            final OWLClass parameter = dataFactory.getOWLClass(parameterIri);
400            final IParameterTypeVisitor visitor = new ParameterTypeVisitor();
401            for (final OWLAxiom ax : kisao.getAxioms(AxiomType.SUBCLASS_OF)) {
402                ax.accept(new OWLAxiomVisitorAdapter() {
403                    public void visit(OWLSubClassOfAxiom axiom) {
404                        OWLClassExpression subClz = axiom.getSubClass();
405                        if (subClz != null && (parameter.equals(subClz))) {
406                            OWLClassExpression expression = axiom.getSuperClass();
407                            if (expression.isAnonymous()) {
408                                expression.accept(visitor);
409                            }
410                        }
411                    }
412                });
413            }
414            return visitor.getResult();
415        }

447        public IRI getIRIByNameOrSynonym(String name) {
448            assert name != null;
449            name = normalize(name);
450            IRI iri = label2iri.get(name);
451            if (iri != null) return iri;
452            OWLAnnotationProperty synonym = dataFactory.getOWLAnnotationProperty(SYNONYM_IRI);
453            for (OWLAnnotationAssertionAxiom ax : kisao.getAxioms(AxiomType.ANNOTATION_ASSERTION)) {
454                OWLAnnotation annotation = ax.getAnnotation();
455                if (synonym.equals(annotation.getProperty())
456                        && name.equals(normalize(getLiteral(annotation)))) {
457                    OWLAnnotationSubject subject = ax.getSubject();
458                    if (subject instanceof IRI) {
459                        return (IRI) subject;
460                    }
461                }
462            }
463            return null;
464        }

466        public IRI getIRIbyURN(String id) {
467            if (pattern.matcher(id).find()) {
468                Matcher matcher = idPattern.matcher(id);
469                matcher.find();
470                id = matcher.group();
471            }
472            return IRI.create(String.format("%sKISAO_%s", KISAO_PREFIX, id));
473        }

481            if (iri == null) return null;
482            String fragment = iri.getFragment();

499        private OWLClassExpression getClassExpressionByCharacteristics(OWLClassExpression algorithm, boolean positive, IRI... characteristicIRI) {
500            OWLObjectProperty hasCharacteristic = dataFactory.getOWLObjectProperty(HAS_CHARACTERISTIC_IRI);
501            for (IRI characteristic : characteristicIRI) {
502                OWLObjectSomeValuesFrom someValues = dataFactory.getOWLObjectSomeValuesFrom(hasCharacteristic,
503                        dataFactory.getOWLClass(characteristic));
504                algorithm = dataFactory.getOWLObjectIntersectionOf(algorithm,
505                        positive ? someValues : someValues.getObjectComplementOf());
506            }
507            return algorithm;
508        }

510        private Set<IRI> getPropertyValues(OWLClassExpression algorithm, final IPropertyValueByAlgorithmVisitor visitor) {
511            visitor.clear();
512            final Set<OWLClass> superAlgorithms = getAncestors(algorithm);
513            Set<OWLSubClassOfAxiom> axioms = algorithm instanceof OWLClass ?
514                    kisao.getSubClassAxiomsForSubClass((OWLClass) algorithm) : new HashSet<OWLSubClassOfAxiom>();
515            for (OWLClass clzz : superAlgorithms) {
516                axioms.addAll(kisao.getSubClassAxiomsForSubClass(clzz));
517            }
518            for (final OWLAxiom ax : axioms) {
519                ax.accept(new OWLAxiomVisitorAdapter() {
520                    public void visit(OWLSubClassOfAxiom axiom) {
521                        OWLClassExpression expression = axiom.getSuperClass();
522                        if (expression.isAnonymous()) {
523                            expression.accept(visitor);
524                        }
525                    }
526                });
527            }
528    
529            return visitor.getValues();
530        }

532        private void collectLabels() {
533            for (OWLClass clazz : kisao.getClassesInSignature()) {
534                String label = getAnnotation(clazz, dataFactory.getRDFSLabel());
535                if (label != null) {
536                    label2iri.put(normalize(label), clazz.getIRI());
537                }
538            }
539            for (OWLObjectProperty property : kisao.getObjectPropertiesInSignature()) {
540                IRI iri = property.getIRI();
541                Set<OWLAnnotation> labels = property.getAnnotations(kisao, dataFactory.getRDFSLabel());
542                if (!labels.isEmpty()) {
543                    String label = getLiteral(labels);
544                    if (label != null) {
545                        label2iri.put(normalize(label), iri);
546                    }
547                }
548            }
549        }
550    
551        private String normalize(String label) {
552            return label == null ? label : label.trim().toLowerCase().replace("'", "");
553        }

563        private void addInferredAxioms(OWLOntologyManager manager) {
564            List<InferredAxiomGenerator<? extends OWLAxiom>> generators = Arrays.asList(
565                    (InferredAxiomGenerator<? extends OWLAxiom>)new InferredClassAssertionAxiomGenerator(),
566                    new InferredEquivalentClassAxiomGenerator(),
567                    new InferredInverseObjectPropertiesAxiomGenerator(),
568                    new InferredPropertyAssertionGenerator(),
569                    new InferredSubClassAxiomGenerator(),
570                    new InferredSubObjectPropertyAxiomGenerator()
571            );
572    
573            new InferredOntologyGenerator(reasoner, generators).fillOntology(manager, kisao);
574        }

576        private Set<OWLClass> getAncestors(OWLClassExpression algorithm) {
577            final Set<OWLClass> ancestors = new HashSet<OWLClass>();
578            // not anonymous ancestors
579            for (Node<OWLClass> clazz : reasoner.getSuperClasses(algorithm, false)) {
580                ancestors.add(clazz.getRepresentativeElement());
581            }
582            return ancestors;
583        }

585        private List<IRI> getSubBranch(IRI branchParent) {
586            List<IRI> result = new ArrayList<IRI>();
587            for (Node<OWLClass> node : reasoner.getSubClasses(dataFactory.getOWLClass(branchParent), false)) {
588                result.add(node.getRepresentativeElement().getIRI());
589            }
590            return result;
591        }

593        private List<IRI> getSubBranchLeaves(IRI branchParent) {
594            List<IRI> result = new ArrayList<IRI>();
595            for (Node<OWLClass> node : reasoner.getSubClasses(dataFactory.getOWLClass(branchParent), false)) {
596                OWLClass element = node.getRepresentativeElement();
597                if (element.getSubClasses(kisao).isEmpty() && !element.isOWLNothing()) {
598                    result.add(element.getIRI());
599                }
600            }
601            return result;
602        }

604        private String getAnnotation(OWLClass clazz, OWLAnnotationProperty property) {
605            assert clazz != null;
606            Set<OWLAnnotation> definition = clazz.getAnnotations(kisao, property);
607            if (!definition.isEmpty()) {
608                return getLiteral(definition);
609            }
610            return null;
611        }

613        private OWLAnnotation getAnnotationAnnotation(IRI entityIRI, OWLAnnotation annotatedAnnotation,
614                                                      OWLAnnotationProperty property) {
615            for (OWLAnnotationAssertionAxiom ax : kisao.getAnnotationAssertionAxioms(entityIRI)) {
616                if (ax.getProperty().getIRI().equals(annotatedAnnotation.getProperty().getIRI())
617                        && annotatedAnnotation.getValue().equals(ax.getValue())) {
618                    for (OWLAnnotation annotation : ax.getAnnotations()) {
619                        if (annotation.getProperty().equals(property)) {
620                            return annotation;
621                        }
622                    }
623                    break;
624                }
625            }
626            return null;
627        }
