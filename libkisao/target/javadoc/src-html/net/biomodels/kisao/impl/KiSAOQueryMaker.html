038        /**
039         * Creates a new KiSAOQueryMaker.
040         *
041         * @param kisaoIri IRI for the KiSAO ontology.
042         *                 If kisaoIri == null, then default KiSAO_IRI will be used.
043         * @param factory  OWLReasonerFactory to specify which reasoner should be used.
044         *                 If factory == null, Hermit reasoner will be used.
045         * @param monitor  ReasonerProgressMonitor to be used to listen to the reasoner progress.
046         *                 If monitor == null, NullReasonerProgressMonitor will be used.
047         * @throws OWLOntologyCreationException If something is wrong with the ontology file/url.
048         * @see net.biomodels.kisao.impl.KiSAOQueryMaker KISAO_IRI
049         * @see org.semanticweb.HermiT.Reasoner
050         */
051        public KiSAOQueryMaker(IRI kisaoIri, OWLReasonerFactory factory, ReasonerProgressMonitor monitor) throws OWLOntologyCreationException {
052            OWLOntologyManager manager = OWLManager.createOWLOntologyManager();
053            dataFactory = manager.getOWLDataFactory();
054            if (kisaoIri == null) {
055                kisaoIri = KISAO_IRI;
056            }
057            kisao = manager.loadOntology(kisaoIri);
058    
059            addImportedAxioms(manager);
060    
061            collectLabels();
062    
063            if (factory == null) {
064                factory = new Reasoner.ReasonerFactory();//new FaCTPlusPlusReasonerFactory();
065            }
066            if (monitor == null) {
067                monitor = new NullReasonerProgressMonitor();
068            }
069            reasoner = factory.createNonBufferingReasoner(kisao, new SimpleConfiguration(monitor));
070            addInferredAxioms(manager);
071        }
072    
073        private void addImportedAxioms(OWLOntologyManager manager) {
074            for (OWLOntology importedOntology : kisao.getImports()) {
075                manager.addAxioms(kisao, importedOntology.getAxioms());
076            }
077        }

253        public Set<IRI> getCharacteristics(OWLClassExpression algorithm, final boolean positive, IRI... types) {
254            Set<IRI> characteristics = getPropertyValues(algorithm,
255                    positive ? new PropertyValueByAlgorithmVisitor(HAS_CHARACTERISTIC_IRI) :
256                            new NegativePropertyValueByAlgorithmVisitor(HAS_CHARACTERISTIC_IRI));
257            Set<IRI> result = characteristics;
258            if (types != null && types.length > 0) {
259                result = new HashSet<IRI>();
260                for (IRI characteristic : characteristics) {
261                    for (IRI type : types) {
262                        if (isA(characteristic, type)) {
263                            result.add(characteristic);
264                            break;
265                        }
266                    }
267                }
268            }
269            return result;
270        }

297            if (descendant == null || ancestor == null) return false;
298            return descendant.equals(ancestor) ||
299                    reasoner.isEntailed(dataFactory.getOWLSubClassOfAxiom(descendant, ancestor));

313        public Set<IRI> getDescendants(OWLClassExpression entry, boolean direct) {
314            if (entry == null) return null;
315            final Set<IRI> result = new HashSet<IRI>();
316            // not anonymous descendants
317            for (Node<OWLClass> clazz : reasoner.getSubClasses(entry, direct)) {
318                OWLClass descendant = clazz.getRepresentativeElement();
319                if (descendant != null && !descendant.isOWLNothing()) {
320                    result.add(descendant.getIRI());
321                }
322            }
323            return result;
324        }

344            OWLObjectProperty isHybrid = dataFactory.getOWLObjectProperty(IS_HYBRID_OF_IRI);
345            OWLClass ksa = dataFactory.getOWLClass(KINETIC_SIMULATION_ALGORITHM_IRI);

348            OWLClassExpression query = dataFactory.getOWLObjectIntersectionOf(algorithm, someValuesFrom);
349            return reasoner.getEquivalentClasses(query).contains(algorithm);

363        public Set<IRI> getParameters(OWLClassExpression algorithm) {
364            OWLObjectProperty hasParameter = dataFactory.getOWLObjectProperty(HAS_PARAMETER_IRI);
365            Set<IRI> result = new HashSet<IRI>();
366            for (Node<OWLClass> node : reasoner.getSubClasses(
367                    dataFactory.getOWLClass(KINETIC_SIMULATION_ALGORITHM_PARAMETER_IRI), false)) {
368                OWLClass element = node.getRepresentativeElement();
369                if (element.getSubClasses(kisao).isEmpty() && !element.isOWLNothing()) {
370                    OWLClassExpression query = dataFactory.getOWLObjectSomeValuesFrom(hasParameter, element);
371                    query = dataFactory.getOWLObjectIntersectionOf(algorithm, query);
372                    if (reasoner.isEntailed(dataFactory.getOWLEquivalentClassesAxiom(algorithm, query))) {
373                        result.add(element.getIRI());
374                    }
375                }
376            }
377            return result;
378        }

394        public boolean hasParameter(OWLClassExpression algorithmExpression, final IRI... parameterIRI) {
395            OWLObjectProperty hasParameter = dataFactory.getOWLObjectProperty(HAS_PARAMETER_IRI);
396            for (IRI iri : parameterIRI) {
397                OWLClassExpression query = algorithmExpression;
398                OWLClass parameter = dataFactory.getOWLClass(iri);
399                OWLObjectSomeValuesFrom someValuesFrom = dataFactory.getOWLObjectSomeValuesFrom(hasParameter, parameter);
400                query = dataFactory.getOWLObjectIntersectionOf(query, someValuesFrom);
401                if (!reasoner.isEntailed(dataFactory.getOWLEquivalentClassesAxiom(algorithmExpression, query))) {
402                    return false;
403                }
404            }
405            return true;
406        }

421        public Class getParameterType(IRI parameterIri) {
422            final OWLClass parameter = dataFactory.getOWLClass(parameterIri);
423            final IParameterTypeVisitor visitor = new ParameterTypeVisitor();
424            for (final OWLAxiom ax : kisao.getAxioms(AxiomType.SUBCLASS_OF)) {
425                ax.accept(new OWLAxiomVisitorAdapter() {
426                    public void visit(OWLSubClassOfAxiom axiom) {
427                        OWLClassExpression subClz = axiom.getSubClass();
428                        if (subClz != null && (parameter.equals(subClz))) {
429                            OWLClassExpression expression = axiom.getSuperClass();
430                            if (expression.isAnonymous()) {
431                                expression.accept(visitor);
432                            }
433                        }
434                    }
435                });
436            }
437            return visitor.getResult();
438        }

470        public IRI getIRIByNameOrSynonym(String name) {
471            assert name != null;
472            name = normalize(name);
473            IRI iri = label2iri.get(name);
474            if (iri != null) return iri;
475            OWLAnnotationProperty synonym = dataFactory.getOWLAnnotationProperty(SYNONYM_IRI);
476            for (OWLAnnotationAssertionAxiom ax : kisao.getAxioms(AxiomType.ANNOTATION_ASSERTION)) {
477                OWLAnnotation annotation = ax.getAnnotation();
478                if (synonym.equals(annotation.getProperty())
479                        && name.equals(normalize(getLiteral(annotation)))) {
480                    OWLAnnotationSubject subject = ax.getSubject();
481                    if (subject instanceof IRI) {
482                        return (IRI) subject;
483                    }
484                }
485            }
486            return null;
487        }

504            if (iri == null) return null;
505            String fragment = iri.getFragment();

520        public Set<IRI> getAlgorithmsWithSameCharacteristics(IRI algorithm, IRI... type) {
521            Set<IRI> positive = getCharacteristics(algorithm, true, type);
522            OWLClass ksa = dataFactory.getOWLClass(KINETIC_SIMULATION_ALGORITHM_IRI);
523            OWLClassExpression query = getClassExpressionByCharacteristics(ksa, true,
524                    positive.toArray(new IRI[positive.size()]));
525            Set<IRI> negative = getCharacteristics(algorithm, false, type);
526            query = getClassExpressionByCharacteristics(query, false, negative.toArray(new IRI[negative.size()]));
527            Set<IRI> similar = getAlgorithmsByQuery(query);
528            // remove self
529            similar.remove(algorithm);
530            return similar;
531        }

535        private OWLClassExpression getClassExpressionByCharacteristics(OWLClassExpression algorithm, boolean positive, IRI... characteristicIRI) {
536            OWLObjectProperty hasCharacteristic = dataFactory.getOWLObjectProperty(HAS_CHARACTERISTIC_IRI);
537            for (IRI characteristic : characteristicIRI) {
538                OWLObjectSomeValuesFrom someValues = dataFactory.getOWLObjectSomeValuesFrom(hasCharacteristic,
539                        dataFactory.getOWLClass(characteristic));
540                algorithm = dataFactory.getOWLObjectIntersectionOf(algorithm,
541                        positive ? someValues : someValues.getObjectComplementOf());
542            }
543            return algorithm;
544        }

546        private Set<IRI> getPropertyValues(OWLClassExpression algorithm, final IPropertyValueByAlgorithmVisitor visitor) {
547            visitor.clear();
548            final Set<OWLClass> superAlgorithms = getAncestors(algorithm);
549            Set<OWLSubClassOfAxiom> axioms = algorithm instanceof OWLClass ?
550                    kisao.getSubClassAxiomsForSubClass((OWLClass) algorithm) : new HashSet<OWLSubClassOfAxiom>();
551            for (OWLClass clzz : superAlgorithms) {
552                axioms.addAll(kisao.getSubClassAxiomsForSubClass(clzz));
553            }
554            for (final OWLAxiom ax : axioms) {
555                ax.accept(new OWLAxiomVisitorAdapter() {
556                    public void visit(OWLSubClassOfAxiom axiom) {
557                        OWLClassExpression expression = axiom.getSuperClass();
558                        if (expression.isAnonymous()) {
559                            expression.accept(visitor);
560                        }
561                    }
562                });
563            }
564            return visitor.getValues();
565        }

567        private void collectLabels() {
568            for (OWLClass clazz : kisao.getClassesInSignature()) {
569                String label = getAnnotation(clazz, dataFactory.getRDFSLabel());
570                if (label != null) {
571                    label2iri.put(normalize(label), clazz.getIRI());
572                }
573            }
574            for (OWLObjectProperty property : kisao.getObjectPropertiesInSignature()) {
575                IRI iri = property.getIRI();
576                Set<OWLAnnotation> labels = property.getAnnotations(kisao, dataFactory.getRDFSLabel());
577                if (!labels.isEmpty()) {
578                    String label = getLiteral(labels);
579                    if (label != null) {
580                        label2iri.put(normalize(label), iri);
581                    }
582                }
583            }
584        }

586        private String normalize(String label) {
587            return label == null ? label : label.trim().toLowerCase().replace("'", "");
588        }

590        private String getLiteral(Set<OWLAnnotation> labels) {
591            return ((OWLLiteral) labels.iterator().next().getValue()).getLiteral();
592        }

594        private String getLiteral(OWLAnnotation annotation) {
595            return annotation == null ? "" : ((OWLLiteral) annotation.getValue()).getLiteral();
596        }

598        private void addInferredAxioms(OWLOntologyManager manager) {
599            List<InferredAxiomGenerator<? extends OWLAxiom>> generators = Arrays.asList(
600                    (InferredAxiomGenerator<? extends OWLAxiom>)new InferredClassAssertionAxiomGenerator(),
601                    new InferredEquivalentClassAxiomGenerator(),
602                    new InferredInverseObjectPropertiesAxiomGenerator(),
603                    new InferredPropertyAssertionGenerator(),
604                    new InferredSubClassAxiomGenerator(),
605                    new InferredSubObjectPropertyAxiomGenerator()
606            );
607            new InferredOntologyGenerator(reasoner, generators).fillOntology(manager, kisao);
608        }

610        private Set<OWLClass> getAncestors(OWLClassExpression algorithm) {
611            final Set<OWLClass> ancestors = new HashSet<OWLClass>();
612            // not anonymous ancestors
613            for (Node<OWLClass> clazz : reasoner.getSuperClasses(algorithm, false)) {
614                ancestors.add(clazz.getRepresentativeElement());
615            }
616            return ancestors;
617        }

619        private List<IRI> getSubBranch(IRI branchParent) {
620            List<IRI> result = new ArrayList<IRI>();
621            for (Node<OWLClass> node : reasoner.getSubClasses(dataFactory.getOWLClass(branchParent), false)) {
622                OWLClass element = node.getRepresentativeElement();
623                if (!element.isOWLThing() && !element.isOWLNothing()) {
624                    result.add(element.getIRI());
625                }
626            }
627            return result;
628        }

630        private List<IRI> getSubBranchLeaves(IRI branchParent) {
631            List<IRI> result = new ArrayList<IRI>();
632            for (Node<OWLClass> node : reasoner.getSubClasses(dataFactory.getOWLClass(branchParent), false)) {
633                OWLClass element = node.getRepresentativeElement();
634                if (element.getSubClasses(kisao).isEmpty() && !element.isOWLNothing() && !element.isOWLThing()) {
635                    result.add(element.getIRI());
636                }
637            }
638            return result;
639        }

641        private String getAnnotation(OWLClass clazz, OWLAnnotationProperty property) {
642            assert clazz != null;
643            Set<OWLAnnotation> definition = clazz.getAnnotations(kisao, property);
644            if (!definition.isEmpty()) {
645                return getLiteral(definition);
646            }
647            return null;
648        }

650        private OWLAnnotation getAnnotationAnnotation(IRI entityIRI, OWLAnnotation annotatedAnnotation,
651                                                      OWLAnnotationProperty property) {
652            for (OWLAnnotationAssertionAxiom ax : kisao.getAnnotationAssertionAxioms(entityIRI)) {
653                if (ax.getProperty().getIRI().equals(annotatedAnnotation.getProperty().getIRI())
654                        && annotatedAnnotation.getValue().equals(ax.getValue())) {
655                    for (OWLAnnotation annotation : ax.getAnnotations()) {
656                        if (annotation.getProperty().equals(property)) {
657                            return annotation;
658                        }
659                    }
660                    break;
661                }
662            }
663            return null;
664        }
