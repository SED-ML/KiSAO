042        /**
043         * Creates a new KiSAOQueryMaker.
044         *
045         * @param kisaoIri IRI for the KiSAO ontology.
046         *                 If kisaoIri == null, then default KiSAO_IRI will be used.
047         * @param factory  OWLReasonerFactory to specify which reasoner should be used.
048         *                 If factory == null, Hermit reasoner will be used.
049         * @param monitor  ReasonerProgressMonitor to be used to listen to the reasoner progress.
050         *                 If monitor == null, NullReasonerProgressMonitor will be used.
051         * @throws OWLOntologyCreationException If something is wrong with the ontology file/url.
052         * @see net.biomodels.kisao.impl.KiSAOQueryMaker KISAO_IRI
053         * @see org.semanticweb.HermiT.Reasoner
054         */
055        public KiSAOQueryMaker(IRI kisaoIri, OWLReasonerFactory factory, ReasonerProgressMonitor monitor) throws OWLOntologyCreationException {
056            OWLOntologyManager manager = OWLManager.createOWLOntologyManager();
057            dataFactory = manager.getOWLDataFactory();
058            if (kisaoIri == null) {
059                kisaoIri = KISAO_IRI;
060            }
061            kisao = manager.loadOntologyFromOntologyDocument(kisaoIri);
062            collectLabels();
063    
064            if (factory == null) {
065                factory = new Reasoner.ReasonerFactory();//new FaCTPlusPlusReasonerFactory();
066            }
067            if (monitor == null) {
068                monitor = new NullReasonerProgressMonitor();
069            }
070            reasoner = factory.createNonBufferingReasoner(kisao, new SimpleConfiguration(monitor));
071            addInferredAxioms(manager);
072        }

248        public Set<IRI> getCharacteristics(OWLClassExpression algorithm, final boolean positive, IRI... types) {
249            Set<IRI> characteristics = getPropertyValues(algorithm,
250                    positive ? new PropertyValueByAlgorithmVisitor(HAS_CHARACTERISTIC_IRI) :
251                            new NegativePropertyValueByAlgorithmVisitor(HAS_CHARACTERISTIC_IRI));
252            Set<IRI> result = characteristics;
253            if (types != null && types.length > 0) {
254                result = new HashSet<IRI>();
255                for (IRI characteristic : characteristics) {
256                    for (IRI type : types) {
257                        if (isA(characteristic, type)) {
258                            result.add(characteristic);
259                            break;
260                        }
261                    }
262                }
263            }
264            return result;
265        }

292            if (descendant == null || ancestor == null) return false;
293            return descendant.equals(ancestor) ||
294                    reasoner.isEntailed(dataFactory.getOWLSubClassOfAxiom(descendant, ancestor));

308        public Set<IRI> getDescendants(OWLClassExpression entry, boolean direct) {
309            if (entry == null) return null;
310            final Set<IRI> result = new HashSet<IRI>();
311            // not anonymous descendants
312            for (Node<OWLClass> clazz : reasoner.getSubClasses(entry, direct)) {
313                OWLClass descendant = clazz.getRepresentativeElement();
314                if (descendant != null && !descendant.isOWLNothing()) {
315                    result.add(descendant.getIRI());
316                }
317            }
318            return result;
319        }

339            OWLObjectProperty isHybrid = dataFactory.getOWLObjectProperty(IS_HYBRID_OF_IRI);
340            OWLClass ksa = dataFactory.getOWLClass(KINETIC_SIMULATION_ALGORITHM_IRI);

342            OWLClass algorithm = dataFactory.getOWLClass(iri);
343            OWLClassExpression query = dataFactory.getOWLObjectIntersectionOf(algorithm, someValuesFrom);
344            return reasoner.getEquivalentClasses(query).contains(algorithm);

358        public Set<IRI> getParameters(OWLClassExpression algorithm) {
359            OWLObjectProperty hasParameter = dataFactory.getOWLObjectProperty(HAS_PARAMETER_IRI);
360            Set<IRI> result = new HashSet<IRI>();
361            for (Node<OWLClass> node : reasoner.getSubClasses(
362                    dataFactory.getOWLClass(KINETIC_SIMULATION_ALGORITHM_PARAMETER_IRI), false)) {
363                OWLClass element = node.getRepresentativeElement();
364                if (element.getSubClasses(kisao).isEmpty() && !element.isOWLNothing()) {
365                    OWLClassExpression query = dataFactory.getOWLObjectSomeValuesFrom(hasParameter, element);
366                    query = dataFactory.getOWLObjectIntersectionOf(algorithm, query);
367                    if (reasoner.isEntailed(dataFactory.getOWLEquivalentClassesAxiom(algorithm, query))) {
368                        result.add(element.getIRI());
369                    }
370                }
371            }
372            return result;
373        }

389        public boolean hasParameter(OWLClassExpression algorithmExpression, final IRI... parameterIRI) {
390            OWLObjectProperty hasParameter = dataFactory.getOWLObjectProperty(HAS_PARAMETER_IRI);
391            for (IRI iri : parameterIRI) {
392                OWLClassExpression query = algorithmExpression;
393                OWLClass parameter = dataFactory.getOWLClass(iri);
394                OWLObjectSomeValuesFrom someValuesFrom = dataFactory.getOWLObjectSomeValuesFrom(hasParameter, parameter);
395                query = dataFactory.getOWLObjectIntersectionOf(query, someValuesFrom);
396                if (!reasoner.isEntailed(dataFactory.getOWLEquivalentClassesAxiom(algorithmExpression, query))) {
397                    return false;
398                }
399            }
400            return true;
401        }

416        public Class getParameterType(IRI parameterIri) {
417            final OWLClass parameter = dataFactory.getOWLClass(parameterIri);
418            final IParameterTypeVisitor visitor = new ParameterTypeVisitor();
419            for (final OWLAxiom ax : kisao.getAxioms(AxiomType.SUBCLASS_OF)) {
420                ax.accept(new OWLAxiomVisitorAdapter() {
421                    public void visit(OWLSubClassOfAxiom axiom) {
422                        OWLClassExpression subClz = axiom.getSubClass();
423                        if (subClz != null && (parameter.equals(subClz))) {
424                            OWLClassExpression expression = axiom.getSuperClass();
425                            if (expression.isAnonymous()) {
426                                expression.accept(visitor);
427                            }
428                        }
429                    }
430                });
431            }
432            return visitor.getResult();
433        }

465        public IRI getIRIByNameOrSynonym(String name) {
466            assert name != null;
467            name = normalize(name);
468            IRI iri = label2iri.get(name);
469            if (iri != null) return iri;
470            OWLAnnotationProperty synonym = dataFactory.getOWLAnnotationProperty(SYNONYM_IRI);
471            for (OWLAnnotationAssertionAxiom ax : kisao.getAxioms(AxiomType.ANNOTATION_ASSERTION)) {
472                OWLAnnotation annotation = ax.getAnnotation();
473                if (synonym.equals(annotation.getProperty())
474                        && name.equals(normalize(getLiteral(annotation)))) {
475                    OWLAnnotationSubject subject = ax.getSubject();
476                    if (subject instanceof IRI) {
477                        return (IRI) subject;
478                    }
479                }
480            }
481            return null;
482        }

515        public Set<IRI> getAlgorithmsWithSameCharacteristics(IRI algorithm, IRI... type) {
516            Set<IRI> positive = getCharacteristics(algorithm, true, type);
517            OWLClass ksa = dataFactory.getOWLClass(KINETIC_SIMULATION_ALGORITHM_IRI);
518            OWLClassExpression query = getClassExpressionByCharacteristics(ksa, true,
519                    positive.toArray(new IRI[positive.size()]));
520            Set<IRI> negative = getCharacteristics(algorithm, false, type);
521            query = getClassExpressionByCharacteristics(query, false, negative.toArray(new IRI[negative.size()]));
522            Set<IRI> similar = getAlgorithmsByQuery(query);
523            // remove self
524            similar.remove(algorithm);
525            return similar;
526        }

530        private OWLClassExpression getClassExpressionByCharacteristics(OWLClassExpression algorithm, boolean positive, IRI... characteristicIRI) {
531            OWLObjectProperty hasCharacteristic = dataFactory.getOWLObjectProperty(HAS_CHARACTERISTIC_IRI);
532            for (IRI characteristic : characteristicIRI) {
533                OWLObjectSomeValuesFrom someValues = dataFactory.getOWLObjectSomeValuesFrom(hasCharacteristic,
534                        dataFactory.getOWLClass(characteristic));
535                algorithm = dataFactory.getOWLObjectIntersectionOf(algorithm,
536                        positive ? someValues : someValues.getObjectComplementOf());
537            }
538            return algorithm;
539        }

541        private Set<IRI> getPropertyValues(OWLClassExpression algorithm, final IPropertyValueByAlgorithmVisitor visitor) {
542            visitor.clear();
543            final Set<OWLClass> superAlgorithms = getAncestors(algorithm);
544            Set<OWLSubClassOfAxiom> axioms = algorithm instanceof OWLClass ?
545                    kisao.getSubClassAxiomsForSubClass((OWLClass) algorithm) : new HashSet<OWLSubClassOfAxiom>();
546            for (OWLClass clzz : superAlgorithms) {
547                axioms.addAll(kisao.getSubClassAxiomsForSubClass(clzz));
548            }
549            for (final OWLAxiom ax : axioms) {
550                ax.accept(new OWLAxiomVisitorAdapter() {
551                    public void visit(OWLSubClassOfAxiom axiom) {
552                        OWLClassExpression expression = axiom.getSuperClass();
553                        if (expression.isAnonymous()) {
554                            expression.accept(visitor);
555                        }
556                    }
557                });
558            }
559            return visitor.getValues();
560        }

562        private void collectLabels() {
563            for (OWLClass clazz : kisao.getClassesInSignature()) {
564                String label = getAnnotation(clazz, dataFactory.getRDFSLabel());
565                if (label != null) {
566                    label2iri.put(normalize(label), clazz.getIRI());
567                }
568            }
569            for (OWLObjectProperty property : kisao.getObjectPropertiesInSignature()) {
570                IRI iri = property.getIRI();
571                Set<OWLAnnotation> labels = property.getAnnotations(kisao, dataFactory.getRDFSLabel());
572                if (!labels.isEmpty()) {
573                    String label = getLiteral(labels);
574                    if (label != null) {
575                        label2iri.put(normalize(label), iri);
576                    }
577                }
578            }
579        }
580    
581        private String normalize(String label) {
582            return label == null ? label : label.trim().toLowerCase().replace("'", "");
583        }

585        private String getLiteral(Set<OWLAnnotation> labels) {
586            return ((OWLLiteral) labels.iterator().next().getValue()).getLiteral();
587        }

589        private String getLiteral(OWLAnnotation annotation) {
590            return annotation == null ? "" : ((OWLLiteral) annotation.getValue()).getLiteral();
591        }

593        private void addInferredAxioms(OWLOntologyManager manager) {
594            List<InferredAxiomGenerator<? extends OWLAxiom>> generators = Arrays.asList(
595                    (InferredAxiomGenerator<? extends OWLAxiom>)new InferredClassAssertionAxiomGenerator(),
596                    new InferredEquivalentClassAxiomGenerator(),
597                    new InferredInverseObjectPropertiesAxiomGenerator(),
598                    new InferredPropertyAssertionGenerator(),
599                    new InferredSubClassAxiomGenerator(),
600                    new InferredSubObjectPropertyAxiomGenerator()
601            );
602            new InferredOntologyGenerator(reasoner, generators).fillOntology(manager, kisao);
603        }

605        private Set<OWLClass> getAncestors(OWLClassExpression algorithm) {
606            final Set<OWLClass> ancestors = new HashSet<OWLClass>();
607            // not anonymous ancestors
608            for (Node<OWLClass> clazz : reasoner.getSuperClasses(algorithm, false)) {
609                ancestors.add(clazz.getRepresentativeElement());
610            }
611            return ancestors;
612        }

614        private List<IRI> getSubBranch(IRI branchParent) {
615            List<IRI> result = new ArrayList<IRI>();
616            for (Node<OWLClass> node : reasoner.getSubClasses(dataFactory.getOWLClass(branchParent), false)) {
617                OWLClass element = node.getRepresentativeElement();
618                if (!element.isOWLThing() && !element.isOWLNothing()) {
619                    result.add(element.getIRI());
620                }
621            }
622            return result;
623        }

625        private List<IRI> getSubBranchLeaves(IRI branchParent) {
626            List<IRI> result = new ArrayList<IRI>();
627            for (Node<OWLClass> node : reasoner.getSubClasses(dataFactory.getOWLClass(branchParent), false)) {
628                OWLClass element = node.getRepresentativeElement();
629                if (element.getSubClasses(kisao).isEmpty() && !element.isOWLNothing() && !element.isOWLThing()) {
630                    result.add(element.getIRI());
631                }
632            }
633            return result;
634        }

636        private String getAnnotation(OWLClass clazz, OWLAnnotationProperty property) {
637            assert clazz != null;
638            Set<OWLAnnotation> definition = clazz.getAnnotations(kisao, property);
639            if (!definition.isEmpty()) {
640                return getLiteral(definition);
641            }
642            return null;
643        }

645        private OWLAnnotation getAnnotationAnnotation(IRI entityIRI, OWLAnnotation annotatedAnnotation,
646                                                      OWLAnnotationProperty property) {
647            for (OWLAnnotationAssertionAxiom ax : kisao.getAnnotationAssertionAxioms(entityIRI)) {
648                if (ax.getProperty().getIRI().equals(annotatedAnnotation.getProperty().getIRI())
649                        && annotatedAnnotation.getValue().equals(ax.getValue())) {
650                    for (OWLAnnotation annotation : ax.getAnnotations()) {
651                        if (annotation.getProperty().equals(property)) {
652                            return annotation;
653                        }
654                    }
655                    break;
656                }
657            }
658            return null;
659        }
