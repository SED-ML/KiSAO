001    package net.biomodels.kisao;
002    
003    import org.semanticweb.owlapi.model.IRI;
004    import org.semanticweb.owlapi.model.OWLClassExpression;
005    import org.semanticweb.owlapi.model.OWLDataFactory;
006    import org.semanticweb.owlapi.reasoner.OWLReasoner;
007    
008    import java.util.Map;
009    import java.util.Set;
010    
011    /**
012     * Provides methods to query KiSAO.
013     * Example:
014     * <pre>
015     * {@code
016     * // Create KiSAOQueryMaker instance, which uses last version of kisao.owl ontology
017     * // (URL: http://biomodels.net/kisao/KISAO).
018     * IKiSAOQueryMaker kisaoQuery = new KiSAOQueryMaker();
019     * // To use kisao.owl, stored locally, instead, one should specify IRI constructor argument:
020     * // IKiSAOQueryMaker kisaoQuery = new KiSAOQueryMaker(IRI.create("file:///..."));
021     *
022     * // Get KiSAO element IRI by name...
023     * IRI iri = kisaoQuery.searchByName("tau-leaping method").iterator().next();
024     * // ... or by MIRIAM URN ...
025     * assert iri.equals(kisaoQuery.searchById("urn:miriam:biomodels.kisao:KISAO_0000039"));
026     * // ... or by ID ...
027     * assert iri.equals(kisaoQuery.searchById("kisao:0000039"));
028     *
029     * // Check if it's an algorithm
030     * assert kisaoQuery.isAlgorithm(iri);
031     * // Get algorithm name
032     * String name = kisaoQuery.getName(iri);
033     * System.out.printf("%s (%s)\n", name, iri);
034     * // Get other names
035     * for (String synonym : kisaoQuery.getAllSynonyms(iri)) {
036     *      System.out.printf(" is also known as %s\n", synonym);
037     * }
038     * // Get definition
039     * System.out.printf(" which is %s\n", kisaoQuery.getDefinition(iri));
040     * // Get references
041     * for (Map.Entry<String, String> entry : kisaoQuery.getLinks(iri).entrySet()) {
042     *      System.out.printf(" (see also: %s -- %s)\n", entry.getKey(), entry.getValue());
043     * }
044     *
045     * System.out.println("");
046     * // Check if algorithm has characteristics discrete variable and stochastic behaviour
047     * assert kisaoQuery.hasCharacteristic(iri, true, KiSAOIRI.STOCHASTIC_SYSTEM_BEHAVIOUR_CHARACTERISTIC_IRI,
048     *      KiSAOIRI.DISCRETE_VARIABLE_CHARACTERISTIC_IRI);
049     * // Get positive algorithm characteristics,
050     * // such as: 'has characteristic' some 'continuous variable'
051     * for (IRI characteristic : kisaoQuery.getCharacteristics(iri, true)) {
052     *      System.out.printf(" has characteristic %s\n", kisaoQuery.getName(characteristic));
053     * }
054     * // Get negative algorithm characteristics,
055     * // such as: not 'has characteristic' some 'spatial description'
056     * for (IRI characteristic : kisaoQuery.getCharacteristics(iri, false)) {
057     *      System.out.printf(" hasn't characteristic %s\n", kisaoQuery.getName(characteristic));
058     * }
059     *
060     * System.out.println("");
061     * // Get parameters
062     * for (IRI parameter : kisaoQuery.getParameters(iri)) {
063     *      System.out.printf(" has parameter %s of type %s\n", kisaoQuery.getName(parameter),
064     *          kisaoQuery.getParameterType(parameter).getName());
065     * }
066     * // Get ancestors (true means that only direct ancestors are included)
067     * for (IRI ancestor : kisaoQuery.getAncestors(iri, true)) {
068     *      System.out.printf(" is a %s\n", kisaoQuery.getName(ancestor));
069     * }
070     *
071     * System.out.println("");
072     * // Get descendants (false means that indirect descendants are also included)
073     * for (IRI descendant : kisaoQuery.getDescendants(iri, false)) {
074     *      System.out.printf(" is specified in %s\n", kisaoQuery.getName(descendant));
075     * }
076     *
077     * System.out.println("\n");
078     * // Get all the algorithms using deterministic rules and continuous variables
079     * for (IRI algorithm : kisaoQuery.getAlgorithmsByCharacteristic(true,
080     *          KiSAOIRI.DETERMINISTIC_SYSTEM_BEHAVIOUR_CHARACTERISTIC_IRI, KiSAOIRI.CONTINUOUS_VARIABLE_CHARACTERISTIC_IRI,
081     *          KiSAOIRI.PROGRESSION_WITH_FIXED_TIME_STEP_CHARACTERISTIC_IRI)) {
082     *      System.out.printf("%s is continuous deterministic and has fixed time steps\n", kisaoQuery.getName(algorithm));
083     * }
084     *
085     * System.out.println("\n");
086     * // Make our own query
087     * OWLReasoner reasoner = kisaoQuery.getReasoner();
088     * OWLDataFactory dataFactory = kisaoQuery.getDataFactory();
089     * System.out.printf("%s and %s are%s disjoint\n",
090     * kisaoQuery.getName(KiSAOIRI.KINETIC_SIMULATION_ALGORITHM_IRI),
091     * kisaoQuery.getName(KiSAOIRI.KINETIC_SIMULATION_ALGORITHM_CHARACTERISTIC_IRI),
092     * reasoner.isEntailed(dataFactory.getOWLDisjointClassesAxiom(
093     *      dataFactory.getOWLClass(KiSAOIRI.KINETIC_SIMULATION_ALGORITHM_IRI),
094     *      dataFactory.getOWLClass(KiSAOIRI.KINETIC_SIMULATION_ALGORITHM_CHARACTERISTIC_IRI))) ?
095     *          "" : " not");
096     *
097     * System.out.println("\n");
098     * //Check which parameters has every deterministic algorithm
099     * for (IRI parameterIRI : kisaoQuery.getParametersByCharacteristic(true, KiSAOIRI.DETERMINISTIC_SYSTEM_BEHAVIOUR_CHARACTERISTIC_IRI)) {
100     *      System.out.printf("every deterministic algorithm uses parameter %s\n",
101     *          kisaoQuery.getName(parameterIRI));
102     * }
103     * }
104     * </pre>
105     *
106     * @author Anna Zhukova
107     *         Date: 13-May-2011
108     *         Time: 10:20:20
109     */
110    
111    public interface IKiSAOQueryMaker {
112    
113        /**
114         * Provides a set of IRIs of simulation algorithms, stored in KiSAO.
115         *
116         * @return set of algorithm IRIs.
117         */
118        Set<IRI> getAllAlgorithms();
119    
120        /**
121         * Provides a set of IRIs of simulation algorithm characteristics, stored in KiSAO.
122         * Subsumption characteristics (e.g. 'type of time steps') are not included.
123         *
124         * @return set of algorithm characteristic IRIs.
125         */
126        Set<IRI> getAllCharacteristics();
127    
128        /**
129         * Provides a set of IRIs of simulation algorithm parameters, stored in KiSAO.
130         * Subsumption parameters (e.g. 'granularity control parameter') are not included.
131         *
132         * @return set of algorithm parameter IRIs.
133         */
134        Set<IRI> getAllParameters();
135    
136        /**
137         * Returns name (rdfs:label) by given KiSAO IRI.
138         *
139         * @param iri KiSAO IRI.
140         * @return name of the KiSAO entity.
141         */
142        String getName(IRI iri);
143    
144        /**
145         * Returns definition (rdfs:comment) by given KiSAO IRI.
146         *
147         * @param iri KiSAO IRI.
148         * @return definition of the KiSAO entity.
149         */
150        String getDefinition(IRI iri);
151    
152        /**
153         * Returns whether the KiSAO entry with the given IRI is deprecated or not.
154         *
155         * @param iri KiSAO IRI.
156         * @return true, if the KiSAO entry with the given IRI is deprecated.
157         */
158        boolean isDeprecated(IRI iri);
159    
160        /**
161         * Returns synonyms for the KiSAO entry with the given IRI.
162         *
163         * @param iri KiSAO IRI.
164         * @return set of synonyms.
165         */
166        Set<String> getAllSynonyms(IRI iri);
167    
168        /**
169         * Returns synonyms of the specified type for the KiSAO entry with the given IRI.
170         *
171         * @param iri  KiSAO IRI.
172         * @param type type of the synonyms to look for.
173         * @return set of synonyms.
174         * @see net.biomodels.kisao.SynonymType
175         */
176        Set<String> getSynonyms(IRI iri, SynonymType type);
177    
178        /**
179         * Returns links to the articles/books, describing the KiSAO entry with the given IRI.
180         *
181         * @param iri KiSAO IRI.
182         * @return map article-location (usually, a MIRIAM URN) -> article-name.
183         */
184        Map<String, String> getLinks(IRI iri);
185    
186        /**
187         * Given a KiSAO IRI of a simulation algorithm,
188         * returns a collection of IRIs of characteristics, it possess.
189         *
190         * @param algorithmIri KiSAO IRI of a simulation algorithm.
191         * @param positive     false, if negated characteristics should be looked at,
192         *                     for example 'KMC' not ('has characteristic' 'spatial description').
193         * @param type         type (optional) IRIs of the characteristic categories (e.g. 'system behaviour') to be considered.
194         *                     All the characteristics are considered if no type is specified.
195         * @return set of IRIs of the algorithm characteristics.
196         */
197        Set<IRI> getCharacteristics(IRI algorithmIri, boolean positive, IRI... type);
198    
199        /**
200         * Given the class expression, describing potential algorithm,
201         * returns a collection of IRIs of characteristics, it possess.
202         *
203         * @param algorithm    OWLClassExpression describing algorithm.
204         * @param positive     false, if negated characteristics should be looked at,
205         *                     for example 'KMC' not ('has characteristic' 'spatial description').
206         * @param type         type (optional) IRIs of the characteristic categories (e.g. 'system behaviour') to be considered.
207         *             All the characteristics are considered if no type is specified.
208         * @return set of IRIs of the algorithm characteristics.
209         */
210        Set<IRI> getCharacteristics(OWLClassExpression algorithm, boolean positive, IRI... type);
211    
212        /**
213         * Given KiSAO IRIs of simulation algorithm characteristics,
214         * returns a collection of IRIs of simulation algorithms, which have them.
215         *
216         * @param characteristicIri KiSAO IRIs of simulation algorithm characteristics.
217         * @param positive          false, if negated characteristics should be looked at,
218         *                          for example 'KMC' not ('has characteristic' 'spatial description').
219         * @return set of IRIs of algorithms, which have the characteristic.
220         */
221        Set<IRI> getAlgorithmsByCharacteristic(boolean positive, IRI... characteristicIri);
222    
223        /**
224         * Checks if the algorithm with the specified KiSAO IRI 'has characteristic' (or not 'has characteristic', if positive is false)
225         * with the specified IRIs.
226         * Example:
227         * <pre>
228         * {@code
229         * IKiSAOQueryMaker query = ...
230         * IRI lsoda = IRI.create("http://www.biomodels.net/kisao/KISAO#KISAO_0000094");
231         * // false specifies that we check a negated axiom: "LSODA not ('has characteristic' some 'spatial description')"
232         * assert true ==  query.hasCharacteristic(lsoda, KiSAO.SPATIAL_DESCRIPTION_CHARACTERISTIC_IRI, false);
233         * // true specifies that we check a normal axiom: "LSODA 'has characteristic' some 'continuous variable'"
234         * assert true ==  query.hasCharacteristic(lsoda, KiSAO.CONTINUOUS_VARIABLE_CHARACTERISTIC_IRI, true);
235         * }
236         * </pre>
237         *
238         * @param algorithmIRI      KiSAO IRI of the simulation algorithm.
239         * @param characteristicIRI KiSAO IRIs of the simulation algorithm characteristics.
240         * @param positive          false, if negated characteristics should be looked at.
241         * @return true, if the algorithms with the specified KiSAO IRI 'has characteristic' (or not 'has characteristic', if positive is false)
242         *         with the specified IRI.
243         */
244        boolean hasCharacteristic(IRI algorithmIRI, boolean positive, IRI... characteristicIRI);
245    
246        /**
247         * Returns a set of parameters that are used by the specified algorithm.
248         *
249         * @param algorithm KiSAO IRI of the algorithm.
250         * @return set of parameter KiSAO IRIs.
251         */
252        Set<IRI> getParameters(IRI algorithm);
253    
254        /**
255         * Returns a set of parameters that are used by the class expression, describing potential algorithm.
256         * Example:
257         * <pre>
258         * {@code
259         * OWLDataFactory dataFactory = kisaoQueryMaker.getDataFactory();
260         * OWLClassExpression algorithm = dataFactory.getOWLObjectSomeValuesFrom(dataFactory.getOWLObjectProperty(HAS_CHARACTERISTIC_IRI),
261         * dataFactory.getOWLClass(DETERMINISTIC_SYSTEM_BEHAVIOUR_CHARACTERISTIC_IRI));
262         * // parameters of methods with deterministic behaviour
263         * for (IRI parameter : kisaoQueryMaker.getParameters(algorithm)) {
264         * System.out.println(kisaoQueryMaker.getName(parameter));
265         * }
266         * }
267         * </pre>
268         *
269         * @param algorithm OWLClassExpression describing potential algorithm.
270         * @return set of parameter KiSAO IRIs.
271         */
272        Set<IRI> getParameters(OWLClassExpression algorithm);
273    
274        /**
275         * Returns a set of parameters that are used by the algorithm, which have specified characteristic(s).
276         *
277         * @param characteristicIRI characteristic KiSAO IRIs.
278         * @param positive          false, if negated characteristics should be looked at.
279         * @return set of parameter KiSAO IRIs.
280         */
281        Set<IRI> getParametersByCharacteristic(boolean positive, IRI... characteristicIRI);
282    
283        /**
284         * Returns a set of parameters that are used by the algorithm, which have specified characteristic(s)
285         * and is descendant of the specified algorithm.
286         *
287         * @param ancestor          ancestor algorithm KiSAO IRI.
288         * @param characteristicIRI characteristic KiSAO IRIs.
289         * @param positive          false, if negated characteristics should be looked at.
290         * @return set of parameter KiSAO IRIs.
291         */
292        Set<IRI> getParametersByAncestorAndCharacteristic(IRI ancestor, boolean positive, IRI... characteristicIRI);
293    
294        /**
295         * Returns a set of parameters that are used by the algorithm, which have specified characteristic(s)
296         * and is descendant of the specified algorithm.
297         *
298         * @param ancestor          OWLClassExpression describing ancestor algorithm.
299         * @param characteristicIRI characteristic KiSAO IRIs.
300         * @param positive          false, if negated characteristics should be looked at.
301         * @return set of parameter KiSAO IRIs.
302         */
303        Set<IRI> getParametersByAncestorAndCharacteristic(OWLClassExpression ancestor, boolean positive, IRI... characteristicIRI);
304    
305        /**
306         * Checks if the algorithm with the specified KiSAO IRI 'has parameter' with the specified IRIs.
307         * Example:
308         * <pre>
309         * {@code
310         * IKiSAOQueryMaker kisaoQueryMaker = ...
311         * IRI lsoda = IRI.create("http://www.biomodels.net/kisao/KISAO#KISAO_0000094");
312         * IRI stepSize = IRI.create("http://www.biomodels.net/kisao/KISAO#KISAO_0000114");
313         * assert true ==  kisaoQueryMaker.hasParameter(lsoda, stepSize);
314         * }
315         * </pre>
316         *
317         * @param algorithmIRI KiSAO IRI of the simulation algorithm.
318         * @param parameterIRI KiSAO IRIs of the simulation algorithm parameters.
319         * @return true, if the algorithms with the specified KiSAO IRI 'has parameter' with the specified IRI.
320         */
321        boolean hasParameter(IRI algorithmIRI, IRI... parameterIRI);
322    
323        /**
324         * Checks if the algorithm, described by given OWLClassExpression, 'has parameter' with the specified IRIs.
325         * Example:
326         * <pre>
327         * {@code
328         * IKiSAOQueryMaker kisaoQueryMaker = ...
329         * OWLDataFactory dataFactory = kisaoQueryMaker.getDataFactory();
330         * OWLClassExpression algorithm = dataFactory.getOWLObjectSomeValuesFrom(dataFactory.getOWLObjectProperty(HAS_CHARACTERISTIC_IRI),
331         * dataFactory.getOWLClass(DETERMINISTIC_SYSTEM_BEHAVIOUR_CHARACTERISTIC_IRI));
332         * IRI stepSize = IRI.create("http://www.biomodels.net/kisao/KISAO#KISAO_0000114");
333         * assert true ==  kisaoQueryMaker.hasParameter(algorithm, stepSize);
334         * }
335         * </pre>
336         *
337         * @param algorithm    OWLClassExpression describing algorithm.
338         * @param parameterIRI KiSAO IRIs of the simulation algorithm parameters.
339         * @return true, if the algorithms, described by given OWLClassExpression, 'has parameter' with the specified IRI.
340         */
341        boolean hasParameter(OWLClassExpression algorithm, IRI... parameterIRI);
342    
343        /**
344         * Checks if the algorithm, described by its characteristics, 'has parameter' with the specified IRIs.
345         *
346         * @param characteristicIRIs KiSAO IRIs of the characteristics.
347         * @param parameterIRI       KiSAO IRIs of the simulation algorithm parameters.
348         * @param positive           false, if negated characteristics should be looked at.
349         * @return true, if the algorithms, described by given characteristics, 'has parameter' with the specified IRI.
350         */
351        boolean hasParameter(boolean positive, IRI[] characteristicIRIs, IRI... parameterIRI);
352    
353        /**
354         * Given the KiSAO IRI of a parameter, returns parameter type, e.g. Double for 'absolute tolerance'.
355         *
356         * @param parameter KiSAO IRI of the parameter.
357         * @return Class specifying the parameter type.
358         */
359        Class getParameterType(IRI parameter);
360    
361        /**
362         * Returns type (algorithm, characteristic or parameter) of the KiSAO element with the given IRI.
363         *
364         * @param iri KiSAO IRI.
365         * @return KiSAOType
366         */
367        KiSAOType getType(IRI iri);
368    
369        /**
370         * Checks if the element with the specified IRI represents simulation algorithm in KiSAO.
371         *
372         * @param iri KiSAO IRI.
373         * @return true, if the element with the specified IRI represents simulation algorithm in KiSAO.
374         */
375        boolean isAlgorithm(IRI iri);
376    
377        /**
378         * Checks if the element with the specified IRI represents
379         * simulation algorithm characteristic in KiSAO.
380         *
381         * @param iri KiSAO IRI.
382         * @return true, if the element with the specified IRI represents
383         *         simulation algorithm characteristic in KiSAO.
384         */
385        boolean isCharacteristic(IRI iri);
386    
387        /**
388         * Checks if the element with the specified IRI represents
389         * simulation algorithm parameter in KiSAO.
390         *
391         * @param iri KiSAO IRI.
392         * @return true, if the element with the specified IRI represents
393         *         simulation algorithm parameter in KiSAO.
394         */
395        boolean isParameter(IRI iri);
396    
397        /**
398         * Looks for a KiSAO entry with the specified name or synonym.
399         *
400         * @param name String representing name or synonym.
401         * @return set of KiSAO IRI.
402         */
403        Set<IRI> searchByName(String name);
404    
405        /**
406         * Looks for a KiSAO entries matching specified query.
407         *
408         * @param query String containing entry identifier (e.g. kisao:0000001)
409         *        or MIRIAM URI (e.g. urn:miriam:biomodels.kisao:KISAO_0000001)
410         *        or URI (e.g. "http://www.biomodels.net/kisao/KISAO#KISAO_0000259").
411         * @return KiSAO IRI.
412         */
413        IRI searchById(String query);
414    
415        /**
416         * Returns a set of descendant IRIs for the specified entry.
417         *
418         * @param entryIRI KiSAO IRI.
419         * @param direct   Whether should look only for direct descendants.
420         * @return set of descendant IRIs.
421         */
422        Set<IRI> getDescendants(IRI entryIRI, boolean direct);
423    
424        /**
425         * Returns a set of descendant IRIs for the specified entry.
426         *
427         * @param entry  OWLClassExpression describing ancestor entry.
428         * @param direct Whether should look only for direct descendants.
429         * @return set of descendant IRIs.
430         */
431        Set<IRI> getDescendants(OWLClassExpression entry, boolean direct);
432    
433        /**
434         * Returns a set of ancestor IRIs for the specified entry.
435         *
436         * @param entryIRI KiSAO IRI.
437         * @param direct   Whether should look only for direct ancestors.
438         * @return set of ancestor IRIs.
439         */
440        Set<IRI> getAncestors(IRI entryIRI, boolean direct);
441    
442        /**
443         * Returns a set of ancestor IRIs for the specified entry.
444         *
445         * @param entry  OWLClassExpression describing descendant entry.
446         * @param direct Whether should look only for direct ancestors.
447         * @return set of ancestor IRIs.
448         */
449        Set<IRI> getAncestors(OWLClassExpression entry, boolean direct);
450    
451        /**
452         * Checks if a descendantCandidate entry is a descendant (concerning rdf:subClassOf relationship)
453         * of the ancestorCandidate entry in KiSAO.
454         *
455         * @param descendantCandidate KiSAO IRI of the possible descendant.
456         * @param ancestorCandidate   KiSAO IRI of the possible ancestor.
457         * @return true, if the descendantCandidate entry is a descendant (concerning rdf:subClassOf relationship)
458         *         of the ancestorCandidate entry.
459         */
460        boolean isA(IRI descendantCandidate, IRI ancestorCandidate);
461    
462        /**
463         * Checks if a descendantCandidate entry is a descendant (concerning rdf:subClassOf relationship)
464         * of the ancestorCandidate entry in KiSAO.
465         *
466         * @param descendantCandidate OWLClassExpression for descendant.
467         * @param ancestorCandidate   OWLClassExpression for ancestor.
468         * @return true, if the descendantCandidate entry is a descendant (concerning rdf:subClassOf relationship)
469         *         of the ancestorCandidate entry.
470         */
471        boolean isA(OWLClassExpression descendantCandidate, OWLClassExpression ancestorCandidate);
472    
473    
474        /**
475         * Returns miriam urn (urn:miriam:biomodels.kisao:KISAO_XXXXXXX) by KiSAO IRI.
476         *
477         * @param iri KiSAO IRI.
478         * @return string, representing miriam urn (urn:miriam:biomodels.kisao:KISAO_XXXXXXX)
479         */
480        String getMiriamURI(IRI iri);
481    
482        /**
483         * Returns id (kisao:XXXXXXX) by KiSAO IRI.
484         *
485         * @param iri KiSAO IRI.
486         * @return string, representing id (kisao:XXXXXXX)
487         */
488        String getId(IRI iri);
489    
490        /**
491         * Checks whether the algorithm with the specified IRI is a hybrid one.
492         *
493         * @param iri KiSAO IRI.
494         * @return if algorithm with the specified IRI is a hybrid one.
495         */
496        boolean isHybrid(IRI iri);
497    
498        /**
499         * Returns IRIs of the algorithms, the specified one is a hybrid of.
500         *
501         * @param iri KiSAO IRI of the hybrid algorithm.
502         * @return set of IRIs of the algorithms, the specified one is a hybrid of or null if it's not hybrid.
503         */
504        Set<IRI> getHybridOf(IRI iri);
505    
506        /**
507         * Returns reasoner.
508         *
509         * @return OWLReasoner.
510         */
511        OWLReasoner getReasoner();
512    
513        /**
514         * Returns data factory.
515         *
516         * @return OWLDataFactory.
517         */
518        OWLDataFactory getDataFactory();
519    
520        /**
521         * Returns algorithms having the same characteristics as the specified one.
522         * @param algorithm The sample algorithm IRI.
523         * @param type (optional) IRIs of the characteristic categories (e.g. 'system behaviour') to be considered.
524         *             All the characteristics are considered if no type is specified.
525         * @return set of algorithm IRIs.
526         */
527        Set<IRI> getAlgorithmsWithSameCharacteristics(IRI algorithm, IRI... type);
528    
529        /**
530         * Returns algorithms described by the specified query.
531         * @param query OWLClassExpression describing algorithm.
532         * @return set of algorithm IRIS.
533         */
534        Set<IRI> getAlgorithmsByQuery(OWLClassExpression query);
535    }
