001    package net.biomodels.kisao;
002    
003    /**
004    * @author Anna Zhukova
005    *         Date: 13-May-2011
006    *         Time: 13:08:40
007    */
008    public enum SynonymType {
009        EXACT("EXACT"),
010        RELATED("RELATED"),
011        NARROW("NARROW"),
012        BROAD("BROAD");
013    
014        private final String name;
015    
016        SynonymType(String name) {
017            this.name = name;
018        }
019    
020        public String getName() {
021            return name;
022        }
023    
024        public String toString() {
025            return name;
026        }
027    
028        public static SynonymType getDefault() {
029            return RELATED;
030        }
031    
032        public static SynonymType byName(String name) {
033            if (name == null) return getDefault();
034            for (SynonymType type : SynonymType.values()) {
035                if (type.getName().equals(name)) return type;
036            }
037            return getDefault();
038        }
039    }
