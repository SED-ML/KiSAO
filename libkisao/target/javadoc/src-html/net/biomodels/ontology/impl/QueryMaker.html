032        /**
033         * Creates a new KiSAOQueryMaker.
034         *
035         * @param ontologyIri IRI for the ontology.
036         * @param factory  OWLReasonerFactory to specify which reasoner should be used.
037         *                 If factory == null, Hermit reasoner will be used.
038         * @param monitor  ReasonerProgressMonitor to be used to listen to the reasoner progress.
039         *                 If monitor == null, NullReasonerProgressMonitor will be used.
040         * @throws org.semanticweb.owlapi.model.OWLOntologyCreationException If something is wrong with the ontology file/url.
041         *
042         * @see org.semanticweb.HermiT.Reasoner
043         */
044        public QueryMaker(IRI ontologyIri, OWLReasonerFactory factory, ReasonerProgressMonitor monitor) throws OWLOntologyCreationException {
045            OWLOntologyManager manager = OWLManager.createOWLOntologyManager();
046            dataFactory = manager.getOWLDataFactory();
047            if (ontologyIri == null) {
048                throw new IllegalArgumentException("Ontology IRI can not be null");
049            }
050            ontology = manager.loadOntology(ontologyIri);
051            addImportedAxioms(manager);
052            if (factory == null) {
053                factory = new Reasoner.ReasonerFactory();//new FaCTPlusPlusReasonerFactory();
054            }
055            if (monitor == null) {
056                monitor = new NullReasonerProgressMonitor();
057            }
058            reasoner = factory.createNonBufferingReasoner(ontology, new SimpleConfiguration(monitor));
059            addInferredAxioms(manager);
060        }

062        public String getAnnotation(OWLEntity entity, OWLAnnotationProperty property) {
063            assert entity != null;
064            Set<OWLAnnotation> annotations = entity.getAnnotations(ontology, property);
065            if (!annotations.isEmpty()) {
066                return getLiteral(annotations.iterator().next());
067            }
068            return null;
069        }

073            Set<OWLAnnotation> annotations = entity.getAnnotations(ontology, dataFactory.getOWLAnnotationProperty(property));
074            if (!annotations.isEmpty()) {
075                return getLiteral(annotations.iterator().next());
076            }
077            return null;

081        public OWLAnnotation getAnnotationAnnotation(IRI entityIRI, OWLAnnotation annotatedAnnotation,
082                                                      OWLAnnotationProperty property) {
083            for (OWLAnnotationAssertionAxiom ax : ontology.getAnnotationAssertionAxioms(entityIRI)) {
084                if (ax.getProperty().getIRI().equals(annotatedAnnotation.getProperty().getIRI())
085                        && annotatedAnnotation.getValue().equals(ax.getValue())) {
086                    for (OWLAnnotation annotation : ax.getAnnotations()) {
087                        if (annotation.getProperty().equals(property)) {
088                            return annotation;
089                        }
090                    }
091                    break;
092                }
093            }
094            return null;
095        }

164         public Set<String> getSynonyms(IRI iri) {
165            OWLClass clazz = dataFactory.getOWLClass(iri);
166            Set<String> result = new HashSet<String>();
167            Set<OWLAnnotation> synonym = clazz.getAnnotations(ontology,
168                    dataFactory.getOWLAnnotationProperty(synonymIRI));
169            for (OWLAnnotation name : synonym) {
170                String syn = getLiteral(name);
171                if (syn != null) {
172                    result.add(syn);
173                }
174            }
175            return result;
176        }

178        public Set<String> getSynonyms(IRI iri, String type) {
179            assert type != null;
180            OWLClass clazz = dataFactory.getOWLClass(iri);
181            Set<String> result = new HashSet<String>();
182            Set<OWLAnnotation> synonym = clazz.getAnnotations(ontology,
183                    dataFactory.getOWLAnnotationProperty(synonymIRI));
184            OWLAnnotationProperty synonymType = dataFactory.getOWLAnnotationProperty(synonymTypeIRI);
185            for (OWLAnnotation name : synonym) {
186                if (type.equals(getLiteral(getAnnotationAnnotation(iri, name, synonymType)))) {
187                    result.add(getLiteral(name));
188                }
189            }
190            return result;
191        }

193        public Map<String, String> getLinks(IRI iri) {
194            OWLClass clazz = dataFactory.getOWLClass(iri);
195            Map<String, String> result = new HashMap<String, String>();
196            Set<OWLAnnotation> seeAlso = clazz.getAnnotations(ontology, dataFactory.getOWLAnnotationProperty(linkIRI));
197            for (OWLAnnotation link : seeAlso) {
198                String description = getLiteral(getAnnotationAnnotation(iri, link,
199                        dataFactory.getOWLAnnotationProperty(linkCommentIRI)));
200                result.put(getLiteral(link), description);
201            }
202            return result;
203        }

205        public Set<IRI> getPropertyValues(OWLClassExpression subject, OWLPropertyExpression property, final boolean positive, IRI... types) {
206            Set<IRI> propertyValues = getPropertyValues(subject,
207                    positive ? new PropertyValueBySubjectVisitor(property) :
208                            new NegativePropertyValueBySubjectVisitor(property));
209            Set<IRI> result = propertyValues;
210            if (types != null && types.length > 0) {
211                result = new HashSet<IRI>();
212                for (IRI iri : propertyValues) {
213                    for (IRI type : types) {
214                        if (isA(iri, type)) {
215                            result.add(iri);
216                            break;
217                        }
218                    }
219                }
220            }
221            return result;
222        }

230                    positive, objects);
231            return getByQuery(query);

234        public OWLClassExpression getPropertySubjectExpression(OWLClassExpression ancestor,
235                                                               OWLObjectPropertyExpression property, boolean positive,
236                                                               IRI... objects) {
237            for (IRI obj : objects) {
238                OWLObjectSomeValuesFrom someValues = dataFactory.getOWLObjectSomeValuesFrom(property,
239                        dataFactory.getOWLClass(obj));
240                ancestor = dataFactory.getOWLObjectIntersectionOf(ancestor,
241                        positive ? someValues : someValues.getObjectComplementOf());
242            }
243            return ancestor;
244        }

246        public boolean hasPropertyValue(OWLClassExpression subject, OWLObjectProperty property, boolean positive,
247                                        final IRI... objects) {
248            OWLClassExpression query = getPropertySubjectExpression(
249                    subject, property, positive, objects);
250            return reasoner.isEntailed(dataFactory.getOWLEquivalentClassesAxiom(query, subject));
251        }

253        public Set<IRI> getByQuery(OWLClassExpression query) {
254            final Set<IRI> result = new HashSet<IRI>();
255            for (OWLClass expression : reasoner.getSubClasses(query, false).getFlattened()) {
256                if (!expression.isOWLNothing()) {
257                    result.add(expression.getIRI());
258                }
259            }
260            return result;
261        }

263        public boolean isA(IRI descendantCandidate, IRI ancestorCandidate) {
264            if (descendantCandidate == null || ancestorCandidate == null) return false;
265            OWLClass descendant = dataFactory.getOWLClass(descendantCandidate);
266            OWLClass ancestor = dataFactory.getOWLClass(ancestorCandidate);
267            return isA(descendant, ancestor);
268        }
269    
270        public boolean isA(OWLClassExpression descendantCandidate, OWLClassExpression ancestorCandidate) {
271            return !(descendantCandidate == null || ancestorCandidate == null)
272                    && (descendantCandidate.equals(ancestorCandidate)
273                        || reasoner.isEntailed(dataFactory.getOWLSubClassOfAxiom(descendantCandidate, ancestorCandidate)));
274        }

280        public Set<IRI> getDescendants(OWLClassExpression entry, boolean direct) {
281            if (entry == null) return null;
282            final Set<IRI> result = new HashSet<IRI>();
283            // not anonymous descendants
284            for (Node<OWLClass> clazz : reasoner.getSubClasses(entry, direct)) {
285                OWLClass descendant = clazz.getRepresentativeElement();
286                if (descendant != null && !descendant.isOWLNothing()) {
287                    result.add(descendant.getIRI());
288                }
289            }
290            return result;
291        }

297        public Set<IRI> getAncestors(OWLClassExpression entry, boolean direct) {
298            if (entry == null) return null;
299            final Set<IRI> result = new HashSet<IRI>();
300            // not anonymous ancestors
301            for (Node<OWLClass> clazz : reasoner.getSuperClasses(entry, direct)) {
302                OWLClass ancestor = clazz.getRepresentativeElement();
303                if (ancestor != null && !ancestor.isOWLThing()) {
304                    result.add(ancestor.getIRI());
305                }
306            }
307            return result;
308        }
309    
310        public Set<OWLClass> getAncestors(OWLClassExpression entry) {
311            final Set<OWLClass> ancestors = new HashSet<OWLClass>();
312            // not anonymous ancestors
313            for (Node<OWLClass> clazz : reasoner.getSuperClasses(entry, false)) {
314                ancestors.add(clazz.getRepresentativeElement());
315            }
316            return ancestors;
317        }

319        public Set<IRI> searchByName(String name) {
320            if (name == null || name.length() <= 0) return null;
321            name = normalize(name);
322            if (name2iri == null) collectNames();
323            Set<IRI> names = name2iri.get(name);
324            return names == null ? null : Collections.unmodifiableSet(names);
325        }

335        public Set<IRI> getSubBranch(IRI branchParent) {
336            Set<IRI> result = new HashSet<IRI>();
337            for (Node<OWLClass> node : reasoner.getSubClasses(dataFactory.getOWLClass(branchParent), false)) {
338                OWLClass element = node.getRepresentativeElement();
339                if (!element.isOWLThing() && !element.isOWLNothing()) {
340                    result.add(element.getIRI());
341                }
342            }
343            return result;
344        }

346        public Set<IRI> getSubBranchLeaves(IRI branchParent) {
347            Set<IRI> result = new HashSet<IRI>();
348            for (Node<OWLClass> node : reasoner.getSubClasses(dataFactory.getOWLClass(branchParent), false)) {
349                OWLClass element = node.getRepresentativeElement();
350                if (element.getSubClasses(ontology).isEmpty() && !element.isOWLNothing() && !element.isOWLThing()) {
351                    result.add(element.getIRI());
352                }
353            }
354            return result;
355        }

415        private void addImportedAxioms(OWLOntologyManager manager) {
416            for (OWLOntology importedOntology : ontology.getImports()) {
417                manager.addAxioms(ontology, importedOntology.getAxioms());
418            }
419        }

421        private void collectNames() {
422            name2iri = new HashMap<String, Set<IRI>>();
423            for (OWLClass clazz : ontology.getClassesInSignature()) {
424                collectAnnotations(clazz, dataFactory.getOWLAnnotationProperty(nameIRI));
425                collectAnnotations(clazz, dataFactory.getOWLAnnotationProperty(synonymIRI));
426            }
427        }
428    
429        private void collectAnnotations(OWLClass clazz, OWLAnnotationProperty property) {
430            for (OWLAnnotation annotation : clazz.getAnnotations(ontology, property)) {
431                String name = getLiteral(annotation);
432                if (name != null) {
433                    name = normalize(name);
434                    Set<IRI> iriSet = name2iri.get(name);
435                    if (iriSet == null) {
436                        iriSet = new HashSet<IRI>();
437                        name2iri.put(name, iriSet);
438                    }
439                    iriSet.add(clazz.getIRI());
440                }
441            }
442        }

444        private void addInferredAxioms(OWLOntologyManager manager) {
445            List<InferredAxiomGenerator<? extends OWLAxiom>> generators = Arrays.asList(
446                    (InferredAxiomGenerator<? extends OWLAxiom>)new InferredClassAssertionAxiomGenerator(),
447                    new InferredEquivalentClassAxiomGenerator(),
448                    new InferredInverseObjectPropertiesAxiomGenerator(),
449                    new InferredPropertyAssertionGenerator(),
450                    new InferredSubClassAxiomGenerator(),
451                    new InferredSubObjectPropertyAxiomGenerator()
452            );
453            new InferredOntologyGenerator(reasoner, generators).fillOntology(manager, ontology);
454        }

456        private Set<IRI> getPropertyValues(OWLClassExpression subject, final IPropertyValueBySubjectVisitor visitor) {
457            visitor.clear();
458            final Set<OWLClass> superSubjects = getAncestors(subject);
459            Set<OWLSubClassOfAxiom> axioms = subject instanceof OWLClass ?
460                    ontology.getSubClassAxiomsForSubClass((OWLClass) subject) : new HashSet<OWLSubClassOfAxiom>();
461            for (OWLClass clzz : superSubjects) {
462                axioms.addAll(ontology.getSubClassAxiomsForSubClass(clzz));
463            }
464            for (final OWLAxiom ax : axioms) {
465                ax.accept(new OWLAxiomVisitorAdapter() {
466                    public void visit(OWLSubClassOfAxiom axiom) {
467                        OWLClassExpression expression = axiom.getSuperClass();
468                        if (expression.isAnonymous()) {
469                            expression.accept(visitor);
470                        }
471                    }
472                });
473            }
474            return visitor.getValues();
475        }
